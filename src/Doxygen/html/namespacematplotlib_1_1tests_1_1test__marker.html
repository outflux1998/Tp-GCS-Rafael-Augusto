170def test_marker_clipping(fig_ref, fig_test):
171 # Plotting multiple markers can trigger different optimized paths in
172 # backends, so compare single markers vs multiple to ensure they are
173 # clipped correctly.
174 marker_count = len(markers.MarkerStyle.markers)
175 marker_size = 50
176 ncol = 7
177 nrow = marker_count // ncol + 1
178
179 width = 2 * marker_size * ncol
180 height = 2 * marker_size * nrow * 2
181 fig_ref.set_size_inches((width / fig_ref.dpi, height / fig_ref.dpi))
182 ax_ref = fig_ref.add_axes([0, 0, 1, 1])
183 fig_test.set_size_inches((width / fig_test.dpi, height / fig_ref.dpi))
184 ax_test = fig_test.add_axes([0, 0, 1, 1])
185
186 for i, marker in enumerate(markers.MarkerStyle.markers):
187 x = i % ncol
188 y = i // ncol * 2
189
190 # Singular markers per call.
191 ax_ref.plot([x, x], [y, y + 1], c='k', linestyle='-', lw=3)
192 ax_ref.plot(x, y, c='k',
193 marker=marker, markersize=marker_size, markeredgewidth=10,
194 fillstyle='full', markerfacecolor='white')
195 ax_ref.plot(x, y + 1, c='k',
196 marker=marker, markersize=marker_size, markeredgewidth=10,
197 fillstyle='full', markerfacecolor='white')
198
199 # Multiple markers in a single call.
200 ax_test.plot([x, x], [y, y + 1], c='k', linestyle='-', lw=3,
201 marker=marker, markersize=marker_size, markeredgewidth=10,
202 fillstyle='full', markerfacecolor='white')
203
204 ax_ref.set(xlim=(-0.5, ncol), ylim=(-0.5, 2 * nrow))
