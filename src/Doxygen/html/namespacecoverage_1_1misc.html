251def substitute_variables(text, variables):
252 """Substitute ``${VAR}`` variables in `text` with their values.
253
254 Variables in the text can take a number of shell-inspired forms::
255
256 $VAR
257 ${VAR}
258 ${VAR?} strict: an error if VAR isn't defined.
259 ${VAR-missing} defaulted: "missing" if VAR isn't defined.
260 $$ just a dollar sign.
261
262 `variables` is a dictionary of variable values.
263
264 Returns the resulting text with values substituted.
265
266 """
267 dollar_pattern = r"""(?x) # Use extended regex syntax
268 \$ # A dollar sign,
269 (?: # then
270 (?P<dollar>\$) | # a dollar sign, or
271 (?P<word1>\w+) | # a plain word, or
272 { # a {-wrapped
273 (?P<word2>\w+) # word,
274 (?:
275 (?P<strict>\?) | # with a strict marker
276 -(?P<defval>[^}]*) # or a default value
277 )? # maybe.
278 }
279 )
280 """
281
282 dollar_groups = ('dollar', 'word1', 'word2')
283
284 def dollar_replace(match):
285 """Called for each $replacement."""
286 # Only one of the dollar_groups will have matched, just get its text.
287 word = next(g for g in match.group(*dollar_groups) if g) # pragma: always breaks
288 if word == "$":
289 return "$"
290 elif word in variables:
291 return variables[word]
292 elif match['strict']:
293 msg = f"Variable {word} is undefined: {text!r}"
294 raise CoverageException(msg)
295 else:
296 return match['defval']
297
298 text = re.sub(dollar_pattern, dollar_replace, text)
299 return text
