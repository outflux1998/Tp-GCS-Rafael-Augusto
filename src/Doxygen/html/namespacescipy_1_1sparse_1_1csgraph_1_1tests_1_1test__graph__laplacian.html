26def _explicit_laplacian(x, normed=False):
27 if sparse.issparse(x):
28 x = x.toarray()
29 x = np.asarray(x)
30 y = -1.0 * x
31 for j in range(y.shape[0]):
32 y[j,j] = x[j,j+1:].sum() + x[j,:j].sum()
33 if normed:
34 d = np.diag(y).copy()
35 d[d == 0] = 1.0
36 y /= d[:,None]**.5
37 y /= d[None,:]**.5
38 return y
