81 # Test different extrapolation modes. See ticket 3557
82 x = np.arange(5, dtype=float)
83 y = x**3
84
85 xp = linspace(-8, 13, 100)
86 xp_zeros = xp.copy()
87 xp_zeros[np.logical_or(xp_zeros < 0., xp_zeros > 4.)] = 0
88 xp_clip = xp.copy()
89 xp_clip[xp_clip < x[0]] = x[0]
90 xp_clip[xp_clip > x[-1]] = x[-1]
91
92 for cls in [UnivariateSpline, InterpolatedUnivariateSpline]:
93 spl = cls(x=x, y=y)
94 for ext in [0, 'extrapolate']:
95 assert_allclose(spl(xp, ext=ext), xp**3, atol=1e-16)
96 assert_allclose(cls(x, y, ext=ext)(xp), xp**3, atol=1e-16)
97 for ext in [1, 'zeros']:
98 assert_allclose(spl(xp, ext=ext), xp_zeros**3, atol=1e-16)
99 assert_allclose(cls(x, y, ext=ext)(xp), xp_zeros**3, atol=1e-16)
100 for ext in [2, 'raise']:
101 assert_raises(ValueError, spl, xp, **dict(ext=ext))
102 for ext in [3, 'const']:
103 assert_allclose(spl(xp, ext=ext), xp_clip**3, atol=1e-16)
104 assert_allclose(cls(x, y, ext=ext)(xp), xp_clip**3, atol=1e-16)
105
106 # also test LSQUnivariateSpline [which needs explicit knots]
107 t = spl.get_knots()[3:4] # interior knots w/ default k=3
108 spl = LSQUnivariateSpline(x, y, t)
109 assert_allclose(spl(xp, ext=0), xp**3, atol=1e-16)
