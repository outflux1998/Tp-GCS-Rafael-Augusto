117def test_no_undesired_distributed_cache_hit(loop):
118 # Dask has a pickle cache for callables that are called many times. Because
119 # the dask backends used to wrap both the functions and the arguments
120 # under instances of the Batch callable class this caching mechanism could
121 # lead to bugs as described in: https://github.com/joblib/joblib/pull/1055
122 # The joblib-dask backend has been refactored to avoid bundling the
123 # arguments as an attribute of the Batch instance to avoid this problem.
124 # This test serves as non-regression problem.
125
126 # Use a large number of input arguments to give the AutoBatchingMixin
127 # enough tasks to kick-in.
128 lists = [[] for _ in range(100)]
129 np = pytest.importorskip('numpy')
130 X = np.arange(int(1e6))
131
132 def isolated_operation(list_, data=None):
133 if data is not None:
134 np.testing.assert_array_equal(data, X)
135 list_.append(uuid4().hex)
136 return list_
137
138 cluster = LocalCluster(n_workers=1, threads_per_worker=2)
139 client = Client(cluster)
140 try:
141 with parallel_backend('dask') as (ba, _):
142 # dispatches joblib.parallel.BatchedCalls
143 res = Parallel()(
144 delayed(isolated_operation)(list_) for list_ in lists
145 )
146

148 # happens in the dask worker process.

152 # isolated_operation so no scattering event should happen under the
