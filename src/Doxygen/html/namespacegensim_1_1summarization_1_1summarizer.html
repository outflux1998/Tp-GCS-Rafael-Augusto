Get a summarized version of the given text.

The output summary will consist of the most representative sentences
and will be returned as a string, divided by newlines.

Note
----
The input should be a string, and must be longer than :const:`~gensim.summarization.summarizer.INPUT_MIN_LENGTH`
sentences for the summary to make sense.
The text will be split into sentences using the split_sentences method in the :mod:`gensim.summarization.texcleaner`
module. Note that newlines divide sentences.

    If both parameters are provided, the ratio will be ignored.

323def summarize_corpus(corpus, ratio=0.2):
324 """Get a list of the most important documents of a corpus using a variation of the TextRank algorithm [1]_.
325 Used as helper for summarize :func:`~gensim.summarization.summarizer.summarizer`
326
327 Note
328 ----
329 The input must have at least :const:`~gensim.summarization.summarizer.INPUT_MIN_LENGTH` documents for the summary
330 to make sense.
331
332
333 Parameters
334 ----------
335 corpus : list of list of (int, int)
336 Given corpus.
337 ratio : float, optional
338 Number between 0 and 1 that determines the proportion of the number of
339 sentences of the original text to be chosen for the summary, optional.
340
341 Returns
342 -------
343 list of str
344 Most important documents of given `corpus` sorted by the document score, highest first.
345
346 """
347 hashable_corpus = _build_hasheable_corpus(corpus)
348
349 # If the corpus is empty, the function ends.
350 if len(corpus) == 0:
351 logger.warning("Input corpus is empty.")
352 return []
353
354 # Warns the user if there are too few documents.
355 if len(corpus) < INPUT_MIN_LENGTH:
356 logger.warning("Input corpus is expected to have at least %d documents.", INPUT_MIN_LENGTH)
357
358 logger.info('Building graph')
359 graph = _build_graph(hashable_corpus)
360
361 logger.info('Filling graph')
362 _set_graph_edge_weights(graph)
363
364 logger.info('Removing unreachable nodes of graph')
365 _remove_unreachable_nodes(graph)
366
367 # Cannot calculate eigenvectors if number of unique documents in corpus < 3.
368 # Warns user to add more text. The function ends.
369 if len(graph.nodes()) < 3:
370 logger.warning("Please add more sentences to the text. The number of reachable nodes is below 3")
371 return []
372
373 logger.info('Pagerank graph')
374 pagerank_scores = _pagerank(graph)
375
376 logger.info('Sorting pagerank scores')
377 hashable_corpus.sort(key=lambda doc: pagerank_scores.get(doc, 0), reverse=True)
378
379 return [list(doc) for doc in hashable_corpus[:int(len(corpus) * ratio)]]
