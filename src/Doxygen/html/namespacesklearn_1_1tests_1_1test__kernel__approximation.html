147def test_skewed_chi2_sampler():
148 # test that RBFSampler approximates kernel on random data
149
150 # compute exact kernel
151 c = 0.03
152 # set on negative component but greater than c to ensure that the kernel
153 # approximation is valid on the group (-c; +\infty) endowed with the skewed
154 # multiplication.
155 Y[0, 0] = -c / 2.0
156
157 # abbreviations for easier formula
158 X_c = (X + c)[:, np.newaxis, :]
159 Y_c = (Y + c)[np.newaxis, :, :]
160
161 # we do it in log-space in the hope that it's more stable
162 # this array is n_samples_x x n_samples_y big x n_features
163 log_kernel = (
164 (np.log(X_c) / 2.0) + (np.log(Y_c) / 2.0) + np.log(2.0) - np.log(X_c + Y_c)
165 )
166 # reduce to n_samples_x x n_samples_y by summing over features in log-space
167 kernel = np.exp(log_kernel.sum(axis=2))
168
169 # approximate kernel mapping
170 transform = SkewedChi2Sampler(skewedness=c, n_components=1000, random_state=42)
171 X_trans = transform.fit_transform(X)
172 Y_trans = transform.transform(Y)
173
174 kernel_approx = np.dot(X_trans, Y_trans.T)
175 assert_array_almost_equal(kernel, kernel_approx, 1)
176 assert np.isfinite(kernel).all(), "NaNs found in the Gram matrix"
177 assert np.isfinite(kernel_approx).all(), "NaNs found in the approximate Gram matrix"
178
179 # test error is raised on when inputs contains values smaller than -c
180 Y_neg = Y.copy()
181 Y_neg[0, 0] = -c * 2.0
182 msg = "X may not contain entries smaller than -skewedness"
183 with pytest.raises(ValueError, match=msg):
184 transform.transform(Y_neg)
