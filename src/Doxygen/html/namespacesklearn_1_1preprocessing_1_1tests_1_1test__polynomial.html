238def test_spline_transformer_periodic_splines_smoothness(degree):
239 """Test that spline transformation is smooth at first / last knot."""
240 X = np.linspace(-2, 10, 10_000)[:, None]
241
242 transformer = SplineTransformer(
243 degree=degree,
244 extrapolation="periodic",
245 knots=[[0.0], [1.0], [3.0], [4.0], [5.0], [8.0]],
246 )
247 Xt = transformer.fit_transform(X)
248
249 delta = (X.max() - X.min()) / len(X)
250 tol = 10 * delta
251
252 dXt = Xt
253 # We expect splines of degree `degree` to be (`degree`-1) times
254 # continuously differentiable. I.e. for d = 0, ..., `degree` - 1 the d-th
255 # derivative should be continuous. This is the case if the (d+1)-th
256 # numerical derivative is reasonably small (smaller than `tol` in absolute
257 # value). We thus compute d-th numeric derivatives for d = 1, ..., `degree`
258 # and compare them to `tol`.
259 #
260 # Note that the 0-th derivative is the function itself, such that we are
261 # also checking its continuity.
262 for d in range(1, degree + 1):
263 # Check continuity of the (d-1)-th derivative
264 diff = np.diff(dXt, axis=0)
265 assert np.abs(diff).max() < tol
266 # Compute d-th numeric derivative
267 dXt = diff / delta
268
269 # As degree `degree` splines are not `degree` times continuously
270 # differentiable at the knots, the `degree + 1`-th numeric derivative
271 # should have spikes at the knots.
272 diff = np.diff(dXt, axis=0)
273 assert np.abs(diff).max() > 1
274
