307def _parse_reveals(file: IO[str]) -> tuple[npt.NDArray[np.str_], list[str]]:
308 """Extract and parse all ``" # E: "`` comments from the passed
309 file-like object.
310
311 All format keys will be substituted for their respective value
312 from `FORMAT_DICT`, *e.g.* ``"{float64}"`` becomes
313 ``"numpy.floating[numpy._typing._64Bit]"``.
314 """
315 string = file.read().replace("*", "")
316
317 # Grab all `# E:`-based comments and matching expressions
318 expression_array, _, comments_array = np.char.partition(
319 string.split("\n"), sep=" # E: "
320 ).T
321 comments = "/n".join(comments_array)
322
323 # Only search for the `{*}` pattern within comments, otherwise
324 # there is the risk of accidentally grabbing dictionaries and sets
325 key_set = set(re.findall(r"\{(.*?)\}", comments))
326 kwargs = {
327 k: FORMAT_DICT.get(k, f"<UNRECOGNIZED FORMAT KEY {k!r}>") for
328 k in key_set
329 }
330 fmt_str = comments.format(**kwargs)
331
332 return expression_array, fmt_str.split("/n")
