634 fit_intercept=False, alpha=alpha, positive=True
635 ).fit(X, y)
636 clf2 = linear_model.Lasso(
637 fit_intercept=False, alpha=alpha, tol=1e-8, positive=True
638 ).fit(X, y)
639 err = linalg.norm(clf1.coef_ - clf2.coef_)
640 assert err < 1e-3
