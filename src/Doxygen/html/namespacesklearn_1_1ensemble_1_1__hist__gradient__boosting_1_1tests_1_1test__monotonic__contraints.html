62def assert_children_values_monotonic(predictor, monotonic_cst):
63 # Make sure siblings values respect the monotonic constraints. Left should
64 # be lower (resp greater) than right child if constraint is POS (resp.
65 # NEG).
66 # Note that this property alone isn't enough to ensure full monotonicity,
67 # since we also need to guanrantee that all the descendents of the left
68 # child won't be greater (resp. lower) than the right child, or its
69 # descendents. That's why we need to bound the predicted values (this is
70 # tested in assert_children_values_bounded)
71 nodes = predictor.nodes
72 left_lower = []
73 left_greater = []
74 for node in nodes:
75 if node["is_leaf"]:
76 continue
77
78 left_idx = node["left"]
79 right_idx = node["right"]
80
81 if nodes[left_idx]["value"] < nodes[right_idx]["value"]:
82 left_lower.append(node)
83 elif nodes[left_idx]["value"] > nodes[right_idx]["value"]:
84 left_greater.append(node)
85
86 if monotonic_cst == MonotonicConstraint.NO_CST:
87 assert left_lower and left_greater
88 elif monotonic_cst == MonotonicConstraint.POS:
89 assert left_lower and not left_greater
90 else: # NEG

145def test_nodes_values(monotonic_cst, seed):
146 # Build a single tree with only one feature, and make sure the nodes
147 # values respect the monotonic constraints.
148
149 # Considering the following tree with a monotonic POS constraint, we
150 # should have:
151 #
152 # root
153 # / \
154 # 5 10 # middle = 7.5
155 # / \ / \
156 # a b c d
157 #
158 # a <= b and c <= d (assert_children_values_monotonic)
159 # a, b <= middle <= c, d (assert_children_values_bounded)
160 # a <= b <= c <= d (assert_leaves_values_monotonic)
161 #
162 # The last one is a consequence of the others, but can't hurt to check
163
164 rng = np.random.RandomState(seed)
165 n_samples = 1000
166 n_features = 1
167 X_binned = rng.randint(0, 255, size=(n_samples, n_features), dtype=np.uint8)
168 X_binned = np.asfortranarray(X_binned)
169
170 gradients = rng.normal(size=n_samples).astype(G_H_DTYPE)
171 hessians = np.ones(shape=1, dtype=G_H_DTYPE)
172
173 grower = TreeGrower(
174 X_binned, gradients, hessians, monotonic_cst=[monotonic_cst], shrinkage=0.1
175 )
176 grower.grow()
177
178 # grow() will shrink the leaves values at the very end. For our comparison
179 # tests, we need to revert the shrinkage of the leaves, else we would
180 # compare the value of a leaf (shrunk) with a node (not shrunk) and the
181 # test would not be correct.
182 for leave in grower.finalized_leaves:
183 leave.value /= grower.shrinkage
184
185 # We pass undefined binning_thresholds because we won't use predict anyway
186 predictor = grower.make_predictor(
187 binning_thresholds=np.zeros((X_binned.shape[1], X_binned.max() + 1))
188 )
189
190 # The consistency of the bounds can only be checked on the tree grower
191 # as the node bounds are not copied into the predictor tree. The
192 # consistency checks on the values of node children and leaves can be
193 # done either on the grower tree or on the predictor tree. We only
194 # do those checks on the predictor tree as the latter is derived from
195 # the former.
196 assert_children_values_monotonic(predictor, monotonic_cst)
197 assert_children_values_bounded(grower, monotonic_cst)
198 assert_leaves_values_monotonic(predictor, monotonic_cst)

202def test_predictions(global_random_seed, use_feature_names):
203 # Train a model with a POS constraint on the first feature and a NEG
204 # constraint on the second feature, and make sure the constraints are
205 # respected by checking the predictions.
206 # test adapted from lightgbm's test_monotone_constraint(), itself inspired
207 # by https://xgboost.readthedocs.io/en/latest/tutorials/monotonic.html
208
209 rng = np.random.RandomState(global_random_seed)
210
211 n_samples = 1000
212 f_0 = rng.rand(n_samples) # positive correlation with y
213 f_1 = rng.rand(n_samples) # negative correslation with y
214 X = np.c_[f_0, f_1]
215 if use_feature_names:
216 pd = pytest.importorskip("pandas")
217 X = pd.DataFrame(X, columns=["f_0", "f_1"])
218
219 noise = rng.normal(loc=0.0, scale=0.01, size=n_samples)
220 y = 5 * f_0 + np.sin(10 * np.pi * f_0) - 5 * f_1 - np.cos(10 * np.pi * f_1) + noise
221
222 if use_feature_names:
223 monotonic_cst = {"f_0": +1, "f_1": -1}
224 else:
225 monotonic_cst = [+1, -1]
226
227 gbdt = HistGradientBoostingRegressor(monotonic_cst=monotonic_cst)
228 gbdt.fit(X, y)
229
230 linspace = np.linspace(0, 1, 100)
231 sin = np.sin(linspace)
232 constant = np.full_like(linspace, fill_value=0.5)
233
234 # We now assert the predictions properly respect the constraints, on each
235 # feature. When testing for a feature we need to set the other one to a
236 # constant, because the monotonic constraints are only a "all else being
237 # equal" type of constraints:
238 # a constraint on the first feature only means that
239 # x0 < x0' => f(x0, x1) < f(x0', x1)
240 # while x1 stays constant.
241 # The constraint does not guanrantee that
242 # x0 < x0' => f(x0, x1) < f(x0', x1')
243
244 # First feature (POS)
245 # assert pred is all increasing when f_0 is all increasing
246 X = np.c_[linspace, constant]
247 pred = gbdt.predict(X)
248 assert is_increasing(pred)
249 # assert pred actually follows the variations of f_0
250 X = np.c_[sin, constant]
251 pred = gbdt.predict(X)
252 assert np.all((np.diff(pred) >= 0) == (np.diff(sin) >= 0))
253
254 # Second feature (NEG)
255 # assert pred is all decreasing when f_1 is all increasing
256 X = np.c_[constant, linspace]
257 pred = gbdt.predict(X)
258 assert is_decreasing(pred)
259 # assert pred actually follows the inverse variations of f_1
260 X = np.c_[constant, sin]
261 pred = gbdt.predict(X)
262 assert ((np.diff(pred) <= 0) == (np.diff(sin) >= 0)).all()
