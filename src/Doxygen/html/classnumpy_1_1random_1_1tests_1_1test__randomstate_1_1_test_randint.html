393 def test_int64_uint64_corner_case(self):
394 # When stored in Numpy arrays, `lbnd` is casted
395 # as np.int64, and `ubnd` is casted as np.uint64.
396 # Checking whether `lbnd` >= `ubnd` used to be
397 # done solely via direct comparison, which is incorrect
398 # because when Numpy tries to compare both numbers,
399 # it casts both to np.float64 because there is
400 # no integer superset of np.int64 and np.uint64. However,
401 # `ubnd` is too large to be represented in np.float64,
402 # causing it be round down to np.iinfo(np.int64).max,
403 # leading to a ValueError because `lbnd` now equals
404 # the new `ubnd`.
405
406 dt = np.int64
407 tgt = np.iinfo(np.int64).max
408 lbnd = np.int64(np.iinfo(np.int64).max)
409 ubnd = np.uint64(np.iinfo(np.int64).max + 1)
410
411 # None of these function calls should
412 # generate a ValueError now.
413 actual = random.randint(lbnd, ubnd, dtype=dt)
414 assert_equal(actual, tgt)
