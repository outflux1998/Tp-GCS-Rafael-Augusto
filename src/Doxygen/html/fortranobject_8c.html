1129 * Returns 0 if the function is successful.

1144 int free_axe = -1;
1145 int i;
1146 npy_intp d;
1147 /* Fill dims where -1 or 0; check dimensions; calc new_size; */

1149 d = PyArray_DIM(arr, i);
1150 if (dims[i] >= 0) {
1151 if (d > 1 && dims[i] != d) {
1152 PyErr_Format(
