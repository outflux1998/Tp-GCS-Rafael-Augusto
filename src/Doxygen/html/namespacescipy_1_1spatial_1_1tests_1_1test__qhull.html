556 # are equal
557
558 facets_1 = set(map(sorted_tuple, facets_1))

562 # The direct check fails for the pathological cases
563 # --- then the convex hull from Delaunay differs (due
564 # to rounding error etc.) from the hull computed
565 # otherwise, by the question whether (tricoplanar)
566 # points that lie almost exactly on the hull are
567 # included as vertices of the hull or not.
568 #
569 # So we check the result, and accept it if the Delaunay
570 # hull line segments are a subset of the usual hull.
571
572 eps = 1000 * np.finfo(float).eps
573
574 for a, b in facets_1:
575 for ap, bp in facets_2:
576 t = points[bp] - points[ap]
577 t /= np.linalg.norm(t) # tangent
578 n = np.array([-t[1], t[0]]) # normal
579
580 # check that the two line segments are parallel
581 # to the same line
582 c1 = np.dot(n, points[b] - points[ap])
583 c2 = np.dot(n, points[a] - points[ap])
584 if not np.allclose(np.dot(c1, n), 0):
585 continue
586 if not np.allclose(np.dot(c2, n), 0):
587 continue
588
589 # Check that the segment (a, b) is contained in (ap, bp)
590 c1 = np.dot(t, points[a] - points[ap])
591 c2 = np.dot(t, points[b] - points[ap])
592 c3 = np.dot(t, points[bp] - points[ap])
593 if c1 < -eps or c1 > c3 + eps:
594 continue

596 continue
