Allocate a new ndarray with aligned memory.

The ndarray is guaranteed *not* aligned to twice the requested alignment.
Eg, if align=4, guarantees it is not aligned to 8. If align=None uses

41def _aligned_zeros(shape, dtype=float, order="C", align=None):
42 """
43 Allocate a new ndarray with aligned memory.
44
45 The ndarray is guaranteed *not* aligned to twice the requested alignment.
46 Eg, if align=4, guarantees it is not aligned to 8. If align=None uses
47 dtype.alignment."""
48 dtype = np.dtype(dtype)
49 if dtype == np.dtype(object):
50 # Can't do this, fall back to standard allocation (which
51 # should always be sufficiently aligned)
52 if align is not None:
53 raise ValueError("object array alignment not supported")
54 return np.zeros(shape, dtype=dtype, order=order)
55 if align is None:
56 align = dtype.alignment
57 if not hasattr(shape, '__len__'):
58 shape = (shape,)
59 size = functools.reduce(operator.mul, shape) * dtype.itemsize
60 buf = np.empty(size + 2*align + 1, np.uint8)
61
62 ptr = buf.__array_interface__['data'][0]
63 offset = ptr % align
64 if offset != 0:
65 offset = align - offset
66 if (ptr % (2*align)) == 0:
67 offset += align
68
69 # Note: slices producing 0-size arrays do not necessarily change
70 # data pointer --- so we use and allocate size+1
71 buf = buf[offset:offset+size+1][:-1]
72 buf.fill(0)
73 data = np.ndarray(shape, dtype, buf, order=order)
74 return data
