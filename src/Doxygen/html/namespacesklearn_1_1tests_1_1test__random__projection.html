152def test_sparse_random_matrix():
153 # Check some statical properties of sparse random matrix
154 n_components = 100
155 n_features = 500
156
157 for density in [0.3, 1.0]:
158 s = 1 / density
159
160 A = _sparse_random_matrix(
161 n_components, n_features, density=density, random_state=0
162 )
163 A = densify(A)
164
165 # Check possible values
166 values = np.unique(A)
167 assert np.sqrt(s) / np.sqrt(n_components) in values
168 assert -np.sqrt(s) / np.sqrt(n_components) in values
169
170 if density == 1.0:
171 assert np.size(values) == 2
172 else:
173 assert 0.0 in values
174 assert np.size(values) == 3
175
176 # Check that the random matrix follow the proper distribution.
177 # Let's say that each element of a_{ij} of A is taken from
178 #
179 # - -sqrt(s) / sqrt(n_components) with probability 1 / 2s
180 # - 0 with probability 1 - 1 / s
181 # - +sqrt(s) / sqrt(n_components) with probability 1 / 2s
182 #
183 assert_almost_equal(np.mean(A == 0.0), 1 - 1 / s, decimal=2)
184 assert_almost_equal(
185 np.mean(A == np.sqrt(s) / np.sqrt(n_components)), 1 / (2 * s), decimal=2
186 )
187 assert_almost_equal(
188 np.mean(A == -np.sqrt(s) / np.sqrt(n_components)), 1 / (2 * s), decimal=2
189 )
190
191 assert_almost_equal(np.var(A == 0.0, ddof=1), (1 - 1 / s) * 1 / s, decimal=2)
192 assert_almost_equal(
193 np.var(A == np.sqrt(s) / np.sqrt(n_components), ddof=1),
194 (1 - 1 / (2 * s)) * 1 / (2 * s),
195 decimal=2,
196 )
197 assert_almost_equal(
198 np.var(A == -np.sqrt(s) / np.sqrt(n_components), ddof=1),
199 (1 - 1 / (2 * s)) * 1 / (2 * s),
200 decimal=2,
