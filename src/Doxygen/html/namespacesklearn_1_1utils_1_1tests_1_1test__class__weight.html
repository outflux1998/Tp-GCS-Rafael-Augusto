77def test_compute_class_weight_invariance():
78 # Test that results with class_weight="balanced" is invariant wrt
79 # class imbalance if the number of samples is identical.
80 # The test uses a balanced two class dataset with 100 datapoints.
81 # It creates three versions, one where class 1 is duplicated
82 # resulting in 150 points of class 1 and 50 of class 0,
83 # one where there are 50 points in class 1 and 150 in class 0,
84 # and one where there are 100 points of each class (this one is balanced
85 # again).
86 # With balancing class weights, all three should give the same model.
87 X, y = make_blobs(centers=2, random_state=0)
88 # create dataset where class 1 is duplicated twice
89 X_1 = np.vstack([X] + [X[y == 1]] * 2)
90 y_1 = np.hstack([y] + [y[y == 1]] * 2)
91 # create dataset where class 0 is duplicated twice
92 X_0 = np.vstack([X] + [X[y == 0]] * 2)
93 y_0 = np.hstack([y] + [y[y == 0]] * 2)
94 # duplicate everything
95 X_ = np.vstack([X] * 2)
96 y_ = np.hstack([y] * 2)
97 # results should be identical
98 logreg1 = LogisticRegression(class_weight="balanced").fit(X_1, y_1)
99 logreg0 = LogisticRegression(class_weight="balanced").fit(X_0, y_0)
100 logreg = LogisticRegression(class_weight="balanced").fit(X_, y_)
101 assert_array_almost_equal(logreg1.coef_, logreg0.coef_)
102 assert_array_almost_equal(logreg.coef_, logreg0.coef_)
