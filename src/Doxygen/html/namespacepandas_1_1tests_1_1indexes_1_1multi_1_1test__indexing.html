865def test_pyint_engine():
866 # GH#18519 : when combinations of codes cannot be represented in 64
867 # bits, the index underlying the MultiIndex engine works with Python
868 # integers, rather than uint64.
869 N = 5
870 keys = [
871 tuple(arr)
872 for arr in [
873 [0] * 10 * N,
874 [1] * 10 * N,
875 [2] * 10 * N,
876 [np.nan] * N + [2] * 9 * N,
877 [0] * N + [2] * 9 * N,
878 [np.nan] * N + [2] * 8 * N + [0] * N,
879 ]
880 ]
881 # Each level contains 4 elements (including NaN), so it is represented
882 # in 2 bits, for a total of 2*N*10 = 100 > 64 bits. If we were using a
883 # 64 bit engine and truncating the first levels, the fourth and fifth
884 # keys would collide; if truncating the last levels, the fifth and
885 # sixth; if rotating bits rather than shifting, the third and fifth.
886
887 for idx in range(len(keys)):
888 index = MultiIndex.from_tuples(keys)
889 assert index.get_loc(keys[idx]) == idx
890
891 expected = np.arange(idx + 1, dtype=np.intp)
892 result = index.get_indexer([keys[i] for i in expected])
893 tm.assert_numpy_array_equal(result, expected)
894
895 # With missing key:
896 idces = range(len(keys))
897 expected = np.array([-1] + list(idces), dtype=np.intp)
898 missing = tuple([0, 1] * 5 * N)
899 result = index.get_indexer([missing] + [keys[i] for i in idces])
900 tm.assert_numpy_array_equal(result, expected)
