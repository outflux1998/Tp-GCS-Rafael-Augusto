408def test_picking_callbacks_overlap(big_on_axes, small_on_axes, click_on):
409 """Test pick events on normal, host or parasite axes."""
410 # Two rectangles are drawn and "clicked on", a small one and a big one
411 # enclosing the small one. The axis on which they are drawn as well as the
412 # rectangle that is clicked on are varied.
413 # In each case we expect that both rectangles are picked if we click on the
