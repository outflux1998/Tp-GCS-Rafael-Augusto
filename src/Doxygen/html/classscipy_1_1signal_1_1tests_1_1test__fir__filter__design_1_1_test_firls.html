501 def test_firls(self):
502 N = 11 # number of taps in the filter
503 a = 0.1 # width of the transition band
504
505 # design a halfband symmetric low-pass filter
506 h = firls(11, [0, a, 0.5-a, 0.5], [1, 1, 0, 0], fs=1.0)
507
508 # make sure the filter has correct # of taps
509 assert_equal(len(h), N)
510
511 # make sure it is symmetric
512 midx = (N-1) // 2
513 assert_array_almost_equal(h[:midx], h[:-midx-1:-1])
514
515 # make sure the center tap is 0.5
516 assert_almost_equal(h[midx], 0.5)
517
518 # For halfband symmetric, odd coefficients (except the center)
519 # should be zero (really small)
520 hodd = np.hstack((h[1:midx:2], h[-midx+1::2]))
521 assert_array_almost_equal(hodd, 0)
522
523 # now check the frequency response
524 w, H = freqz(h, 1)
525 f = w/2/np.pi
526 Hmag = np.abs(H)
527
528 # check that the pass band is close to unity
529 idx = np.logical_and(f > 0, f < a)
530 assert_array_almost_equal(Hmag[idx], 1, decimal=3)
531
532 # check that the stop band is close to zero
533 idx = np.logical_and(f > 0.5-a, f < 0.5)
534 assert_array_almost_equal(Hmag[idx], 0, decimal=3)
