228def test_monotonic_likelihood():

230 # regularization improve monotonically the training set of the bound
231 rng = np.random.RandomState(0)
232 rand_data = RandomData(rng, scale=20)
233 n_components = rand_data.n_components

242 warm_start=True,
243 max_iter=1,
244 random_state=rng,
245 tol=1e-3,
246 )
247 current_lower_bound = -np.infty
248 # Do one training iteration at a time so we can make sure that the
249 # training log likelihood increases after each iteration.
250 for _ in range(600):
251 prev_lower_bound = current_lower_bound
252 current_lower_bound = bgmm.fit(X).lower_bound_
253 assert current_lower_bound >= prev_lower_bound
