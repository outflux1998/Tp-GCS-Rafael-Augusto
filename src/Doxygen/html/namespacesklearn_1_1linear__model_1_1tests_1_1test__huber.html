184def test_huber_better_r2_score():
185 # Test that huber returns a better r2 score than non-outliers"""
186 X, y = make_regression_with_outliers()
187 huber = HuberRegressor(alpha=0.01)
188 huber.fit(X, y)
189 linear_loss = np.dot(X, huber.coef_) + huber.intercept_ - y
190 mask = np.abs(linear_loss) < huber.epsilon * huber.scale_
191 huber_score = huber.score(X[mask], y[mask])
192 huber_outlier_score = huber.score(X[~mask], y[~mask])
193
194 # The Ridge regressor should be influenced by the outliers and hence
195 # give a worse score on the non-outliers as compared to the huber
196 # regressor.
197 ridge = Ridge(alpha=0.01)
198 ridge.fit(X, y)
199 ridge_score = ridge.score(X[mask], y[mask])
200 ridge_outlier_score = ridge.score(X[~mask], y[~mask])
201 assert huber_score > ridge_score
202
203 # The huber model should also fit poorly on the outliers.
204 assert ridge_outlier_score > huber_outlier_score
