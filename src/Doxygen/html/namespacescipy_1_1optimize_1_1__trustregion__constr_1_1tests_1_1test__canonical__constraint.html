9def create_quadratic_function(n, m, rng):
10 a = rng.rand(m)
11 A = rng.rand(m, n)
12 H = rng.rand(m, n, n)
13 HT = np.transpose(H, (1, 2, 0))
14
15 def fun(x):
16 return a + A.dot(x) + 0.5 * H.dot(x).dot(x)
17
18 def jac(x):
19 return A + H.dot(x)
20
21 def hess(x, v):
22 return HT.dot(v)
23
24 return fun, jac, hess
