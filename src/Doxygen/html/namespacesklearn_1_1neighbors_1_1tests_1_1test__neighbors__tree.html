53def compute_kernel_slow(Y, X, kernel, h):
54 d = np.sqrt(((Y[:, None, :] - X) ** 2).sum(-1))
55 norm = kernel_norm(h, X.shape[1], kernel)
56
57 if kernel == "gaussian":
58 return norm * np.exp(-0.5 * (d * d) / (h * h)).sum(-1)
59 elif kernel == "tophat":
60 return norm * (d < h).sum(-1)
61 elif kernel == "epanechnikov":
62 return norm * ((1.0 - (d * d) / (h * h)) * (d < h)).sum(-1)
63 elif kernel == "exponential":
64 return norm * (np.exp(-d / h)).sum(-1)
65 elif kernel == "linear":
66 return norm * ((1 - d / h) * (d < h)).sum(-1)
67 elif kernel == "cosine":
68 return norm * (np.cos(0.5 * np.pi * d / h) * (d < h)).sum(-1)
69 else:
70 raise ValueError("kernel not recognized")
