519 # blocking small arrays and large arrays go through different paths.
520 # the algorithm is triggered depending on the number of element
521 # copies required.
522 # We define a test fixture that forces most tests to go through
523 # both code paths.
524 # Ultimately, this should be removed if a single algorithm is found
525 # to be faster for both small and large arrays.
526 def _block_force_concatenate(arrays):
527 arrays, list_ndim, result_ndim, _ = _block_setup(arrays)
528 return _block_concatenate(arrays, list_ndim, result_ndim)
529
530 def _block_force_slicing(arrays):
531 arrays, list_ndim, result_ndim, _ = _block_setup(arrays)
532 return _block_slicing(arrays, list_ndim, result_ndim)
533
