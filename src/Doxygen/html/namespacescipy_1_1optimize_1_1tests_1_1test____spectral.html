84 # For linear systems F(x) = A x - b = 0, with A positive or

87 def check_solvability(A, b, line_search='cruz'):
88 func = lambda x: A.dot(x) - b
89 xp = np.linalg.solve(A, b)
90 eps = np.linalg.norm(func(xp)) * 1e3

93 assert_(sol.success)
