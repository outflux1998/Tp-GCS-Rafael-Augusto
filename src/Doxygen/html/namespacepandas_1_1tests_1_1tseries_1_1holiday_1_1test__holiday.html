178def test_holidays_within_dates(holiday, start, expected):
179 # see gh-11477
180 #
181 # Fix holiday behavior where holiday.dates returned dates outside
182 # start/end date, or observed rules could not be applied because the
183 # holiday was not in the original date range (e.g., 7/4/2015 -> 7/3/2015).
184 if isinstance(holiday, str):
185 calendar = get_calendar("USFederalHolidayCalendar")
186 holiday = calendar.rule_from_name(holiday)
187
188 assert list(holiday.dates(start, start)) == expected
189
190 # Verify that timezone info is preserved.
191 assert list(
192 holiday.dates(utc.localize(Timestamp(start)), utc.localize(Timestamp(start)))
193 ) == [utc.localize(dt) for dt in expected]
