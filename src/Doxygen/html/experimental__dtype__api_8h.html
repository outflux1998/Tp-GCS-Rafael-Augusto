449 * ********************************
450 * Initialization
451 * ********************************
452 *
453 * Import the experimental API, the version must match the one defined in
454 * the header to ensure changes are taken into account. NumPy will further
455 * runtime-check this.
456 * You must call this function to use the symbols defined in this file.
457 */
458#if !defined(NO_IMPORT) && !defined(NO_IMPORT_ARRAY)
459
460#define __EXPERIMENTAL_DTYPE_VERSION 5
461
462static int
463import_experimental_dtype_api(int version)
464{
465 if (version != __EXPERIMENTAL_DTYPE_VERSION) {
466 PyErr_Format(PyExc_RuntimeError,
467 "DType API version %d did not match header version %d. Please "
468 "update the import statement and check for API changes.",
469 version, __EXPERIMENTAL_DTYPE_VERSION);
470 return -1;
471 }
472 if (__experimental_dtype_api_table != __uninitialized_table) {
473 /* already imported. */
474 return 0;
475 }
476
477 PyObject *multiarray = PyImport_ImportModule("numpy.core._multiarray_umath");
478 if (multiarray == NULL) {
479 return -1;
480 }
481
482 PyObject *api = PyObject_CallMethod(multiarray,
483 "_get_experimental_dtype_api", "i", version);
484 Py_DECREF(multiarray);
485 if (api == NULL) {
486 return -1;
487 }
488 __experimental_dtype_api_table = (void **)PyCapsule_GetPointer(api,
489 "experimental_dtype_api_table");
490 Py_DECREF(api);
491
492 if (__experimental_dtype_api_table == NULL) {
493 __experimental_dtype_api_table = __uninitialized_table;
494 return -1;
495 }
496 return 0;
497}
