143def test_milp_2():
144 # solve MIP with inequality constraints and all integer constraints
145 # source: slide 5,
146 # https://www.cs.upc.edu/~erodri/webpage/cps/theory/lp/milp/slides.pdf
147 # also check that `milp` accepts all valid ways of specifying constraints
148 c = -np.ones(2)
149 A = [[-2, 2], [-8, 10]]
150 b_l = [1, -np.inf]
151 b_u = [np.inf, 13]
152 linear_constraint = LinearConstraint(A, b_l, b_u)
153
154 # solve original problem
155 res1 = milp(c=c, constraints=(A, b_l, b_u), integrality=True)
156 res2 = milp(c=c, constraints=linear_constraint, integrality=True)
157 res3 = milp(c=c, constraints=[(A, b_l, b_u)], integrality=True)
158 res4 = milp(c=c, constraints=[linear_constraint], integrality=True)
159 res5 = milp(c=c, integrality=True,
160 constraints=[(A[:1], b_l[:1], b_u[:1]),
161 (A[1:], b_l[1:], b_u[1:])])
162 res6 = milp(c=c, integrality=True,
163 constraints=[LinearConstraint(A[:1], b_l[:1], b_u[:1]),
164 LinearConstraint(A[1:], b_l[1:], b_u[1:])])
165 res7 = milp(c=c, integrality=True,
166 constraints=[(A[:1], b_l[:1], b_u[:1]),
