513 def test_nested_arraylikes(self, arraylike):
514 # We try storing an array like into an array, but the array-like
515 # will have too many dimensions. This means the shape discovery
516 # decides that the array-like must be treated as an object (a special
517 # case of ragged discovery). The result will be an array with one
518 # dimension less than the maximum dimensions, and the array being
519 # assigned to it (which does work for object or if `float(arraylike)`
520 # works).
521 initial = arraylike(np.ones((1, 1)))
522
523 nested = initial
524 for i in range(np.MAXDIMS - 1):
525 nested = [nested]
526
527 with pytest.raises(ValueError, match=".*would exceed the maximum"):
528 # It will refuse to assign the array into
529 np.array(nested, dtype="float64")
530
531 # If this is object, we end up assigning a (1, 1) array into (1,)
532 # (due to running out of dimensions), this is currently supported but
533 # a special case which is not ideal.
534 arr = np.array(nested, dtype=object)
535 assert arr.shape == (1,) * np.MAXDIMS
536 assert arr.item() == np.array(initial).item()
