134def test_asymmetric_error(quantile, default_solver):
135 """Test quantile regression for asymmetric distributed targets."""
136 n_samples = 1000
137 rng = np.random.RandomState(42)
138 X = np.concatenate(
139 (
140 np.abs(rng.randn(n_samples)[:, None]),
141 -rng.randint(2, size=(n_samples, 1)),
142 ),
143 axis=1,
144 )
145 intercept = 1.23
146 coef = np.array([0.5, -2])
147 # Take care that X @ coef + intercept > 0
148 assert np.min(X @ coef + intercept) > 0
149 # For an exponential distribution with rate lambda, e.g. exp(-lambda * x),
150 # the quantile at level q is:
151 # quantile(q) = - log(1 - q) / lambda
152 # scale = 1/lambda = -quantile(q) / log(1 - q)
153 y = rng.exponential(
154 scale=-(X @ coef + intercept) / np.log(1 - quantile), size=n_samples
155 )
156 model = QuantileRegressor(
157 quantile=quantile,
158 alpha=0,
159 solver=default_solver,
160 ).fit(X, y)
161 # This test can be made to pass with any solver but in the interest
162 # of sparing continuous integration resources, the test is performed
163 # with the fastest solver only.
164
165 assert model.intercept_ == approx(intercept, rel=0.2)
166 assert_allclose(model.coef_, coef, rtol=0.6)
167 assert_allclose(np.mean(model.predict(X) > y), quantile, atol=1e-2)
168
169 # Now compare to Nelder-Mead optimization with L1 penalty
170 alpha = 0.01
171 model.set_params(alpha=alpha).fit(X, y)
172 model_coef = np.r_[model.intercept_, model.coef_]
173
174 def func(coef):
175 loss = mean_pinball_loss(y, X @ coef[1:] + coef[0], alpha=quantile)
176 L1 = np.sum(np.abs(coef[1:]))
177 return loss + alpha * L1
178
179 res = minimize(
180 fun=func,
181 x0=[1, 0, -1],
182 method="Nelder-Mead",
183 tol=1e-12,
184 options={"maxiter": 2000},
185 )
186
187 assert func(model_coef) == approx(func(res.x))
188 assert_allclose(model.intercept_, res.x[0])
189 assert_allclose(model.coef_, res.x[1:])
190 assert_allclose(np.mean(model.predict(X) > y), quantile, atol=1e-2)

194def test_equivariance(quantile, default_solver):
195 """Test equivariace of quantile regression.
196
197 See Koenker (2005) Quantile Regression, Chapter 2.2.3.
198 """
199 rng = np.random.RandomState(42)
200 n_samples, n_features = 100, 5
201 X, y = make_regression(
202 n_samples=n_samples,
203 n_features=n_features,
204 n_informative=n_features,
205 noise=0,
206 random_state=rng,
207 shuffle=False,
208 )
209 # make y asymmetric
210 y += rng.exponential(scale=100, size=y.shape)
211 params = dict(alpha=0, solver=default_solver)
212 model1 = QuantileRegressor(quantile=quantile, **params).fit(X, y)
213
214 # coef(q; a*y, X) = a * coef(q; y, X)
215 a = 2.5
216 model2 = QuantileRegressor(quantile=quantile, **params).fit(X, a * y)
217 assert model2.intercept_ == approx(a * model1.intercept_, rel=1e-5)
218 assert_allclose(model2.coef_, a * model1.coef_, rtol=1e-5)
219
220 # coef(1-q; -a*y, X) = -a * coef(q; y, X)
221 model2 = QuantileRegressor(quantile=1 - quantile, **params).fit(X, -a * y)
222 assert model2.intercept_ == approx(-a * model1.intercept_, rel=1e-5)
223 assert_allclose(model2.coef_, -a * model1.coef_, rtol=1e-5)
224
225 # coef(q; y + X @ g, X) = coef(q; y, X) + g
226 g_intercept, g_coef = rng.randn(), rng.randn(n_features)
227 model2 = QuantileRegressor(quantile=quantile, **params)
228 model2.fit(X, y + X @ g_coef + g_intercept)
229 assert model2.intercept_ == approx(model1.intercept_ + g_intercept)
230 assert_allclose(model2.coef_, model1.coef_ + g_coef, rtol=1e-6)
231
232 # coef(q; y, X @ A) = A^-1 @ coef(q; y, X)
233 A = rng.randn(n_features, n_features)
234 model2 = QuantileRegressor(quantile=quantile, **params)
235 model2.fit(X @ A, y)
236 assert model2.intercept_ == approx(model1.intercept_, rel=1e-5)
237 assert_allclose(model2.coef_, np.linalg.solve(A, model1.coef_), rtol=1e-5)
