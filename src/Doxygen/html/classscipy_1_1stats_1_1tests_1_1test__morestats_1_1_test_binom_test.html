864 # First test for the case where k > mode of PMF
865 i = np.arange(np.ceil(p * n), n+1)
866 d = stats.binom.pmf(k, n, p)
867 # Old way of calculating y, probably consistent with R.
868 y1 = np.sum(stats.binom.pmf(i, n, p) <= d, axis=0)
869 # New way with binary search.
870 ix = _binary_search_for_binom_tst(lambda x1:
871 -stats.binom.pmf(x1, n, p),
872 -d, np.ceil(p * n), n)
873 y2 = n - ix + int(d == stats.binom.pmf(ix, n, p))
874 assert_allclose(y1, y2, rtol=1e-9)
