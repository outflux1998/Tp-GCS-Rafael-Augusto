Detailed Description

Manages the communication between this process and the worker processes.

The manager is run in a local thread.

Args:
    executor: A reference to the ProcessPoolExecutor that owns
        this thread. A weakref will be own by the manager as well as
        references to internal objects used to introspect the state of
        the executor.

783 def shutdown_workers(self):
784 # shutdown all workers in self.processes
785
786 # Create a list to avoid RuntimeError due to concurrent modification of
787 # processes. nb_children_alive is thus an upper bound. Also release the
788 # processes' _worker_exit_lock to accelerate the shutdown procedure, as
789 # there is no need for hand-shake here.
790 with self.processes_management_lock:
791 n_children_to_stop = 0
792 for p in list(self.processes.values()):
793 mp.util.debug(f"releasing worker exit lock on {p.name}")
794 p._worker_exit_lock.release()
795 n_children_to_stop += 1
796
797 mp.util.debug(f"found {n_children_to_stop} processes to stop")
798
799 # Send the right number of sentinels, to make sure all children are
800 # properly terminated. Do it with a mechanism that avoid hanging on
801 # Full queue when all workers have already been shutdown.
802 n_sentinels_sent = 0
803 cooldown_time = 0.001
804 while (n_sentinels_sent < n_children_to_stop
805 and self.get_n_children_alive() > 0):
806 for _ in range(n_children_to_stop - n_sentinels_sent):
807 try:
808 self.call_queue.put_nowait(None)
809 n_sentinels_sent += 1
810 except queue.Full as e:
811 if cooldown_time > 10.0:
812 raise e
813 mp.util.info(
814 "full call_queue prevented to send all sentinels at "
815 "once, waiting..."
816 )
817 sleep(cooldown_time)
818 cooldown_time *= 2
819 break
820

601 def wait_result_broken_or_wakeup(self):
602 # Wait for a result to be ready in the result_queue while checking
603 # that all worker processes are still running, or for a wake up
604 # signal send. The wake up signals come either from new tasks being
605 # submitted, from the executor being shutdown/gc-ed, or from the
606 # shutdown of the python interpreter.
607 result_reader = self.result_queue._reader
608 wakeup_reader = self.thread_wakeup._reader
609 readers = [result_reader, wakeup_reader]
610 worker_sentinels = [p.sentinel for p in list(self.processes.values())]
611 ready = wait(readers + worker_sentinels)
612
613 bpe = None
614 is_broken = True
615 result_item = None
616 if result_reader in ready:
617 try:
618 result_item = result_reader.recv()
619 if isinstance(result_item, _RemoteTraceback):
620 bpe = BrokenProcessPool(
621 "A task has failed to un-serialize. Please ensure that"
622 " the arguments of the function are all picklable."
623 )
624 bpe.__cause__ = result_item
625 else:
626 is_broken = False
627 except BaseException as e:
628 bpe = BrokenProcessPool(
629 "A result has failed to un-serialize. Please ensure that "
630 "the objects returned by the function are always "
631 "picklable."
632 )
633 tb = traceback.format_exception(
634 type(e), e, getattr(e, "__traceback__", None))
635 bpe.__cause__ = _RemoteTraceback(''.join(tb))
636
637 elif wakeup_reader in ready:
638 # This is simply a wake-up event that might either trigger putting
639 # more tasks in the queue or trigger the clean up of resources.
640 is_broken = False
641 else:
642 # A worker has terminated and we don't know why, set the state of
643 # the executor as broken
644 exit_codes = ''
645 if sys.platform != "win32":
646 # In Windows, introspecting terminated workers exitcodes seems
647 # unstable, therefore they are not appended in the exception
648 # message.
649 exit_codes = (
650 "\nThe exit codes of the workers are "
651 f"{get_exitcodes_terminated_worker(self.processes)}"
652 )
653 mp.util.debug('A worker unexpectedly terminated. Workers that '
654 'might have caused the breakage: '
655 + str({p.name: p.exitcode
656 for p in list(self.processes.values())
657 if p is not None and p.sentinel in ready}))
658 bpe = TerminatedWorkerError(
659 "A worker process managed by the executor was unexpectedly "
660 "terminated. This could be caused by a segmentation fault "
661 "while calling the function or by an excessive memory usage "
662 "causing the Operating System to kill the worker.\n"
663 f"{exit_codes}"
664 )
665
666 self.thread_wakeup.clear()
667
668 return result_item, is_broken, bpe
