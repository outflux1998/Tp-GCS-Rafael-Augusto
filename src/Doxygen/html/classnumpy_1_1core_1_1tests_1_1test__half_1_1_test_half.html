89 def test_half_conversion_rounding(self, float_t, shift, offset):
90 # Assumes that round to even is used during casting.
91 max_pattern = np.float16(np.finfo(np.float16).max).view(np.uint16)
92
93 # Test all (positive) finite numbers, denormals are most interesting
94 # however:
95 f16s_patterns = np.arange(0, max_pattern+1, dtype=np.uint16)
96 f16s_float = f16s_patterns.view(np.float16).astype(float_t)
97
98 # Shift the values by half a bit up or a down (or do not shift),
99 if shift == "up":
100 f16s_float = 0.5 * (f16s_float[:-1] + f16s_float[1:])[1:]
101 elif shift == "down":
102 f16s_float = 0.5 * (f16s_float[:-1] + f16s_float[1:])[:-1]
103 else:
104 f16s_float = f16s_float[1:-1]
105
106 # Increase the float by a minimal value:
107 if offset == "up":
108 f16s_float = np.nextafter(f16s_float, float_t(np.inf))
109 elif offset == "down":
110 f16s_float = np.nextafter(f16s_float, float_t(-np.inf))
111
112 # Convert back to float16 and its bit pattern:
113 res_patterns = f16s_float.astype(np.float16).view(np.uint16)
114
115 # The above calculations tries the original values, or the exact
116 # mid points between the float16 values. It then further offsets them
117 # by as little as possible. If no offset occurs, "round to even"
118 # logic will be necessary, an arbitrarily small offset should cause
119 # normal up/down rounding always.
120
121 # Calculate the expected pattern:
122 cmp_patterns = f16s_patterns[1:-1].copy()
123
124 if shift == "down" and offset != "up":
125 shift_pattern = -1
126 elif shift == "up" and offset != "down":
127 shift_pattern = 1
128 else:
129 # There cannot be a shift, either shift is None, so all rounding
130 # will go back to original, or shift is reduced by offset too much.
131 shift_pattern = 0
132
133 # If rounding occurs, is it normal rounding or round to even?
134 if offset is None:
135 # Round to even occurs, modify only non-even, cast to allow + (-1)
136 cmp_patterns[0::2].view(np.int16)[...] += shift_pattern
137 else:
138 cmp_patterns.view(np.int16)[...] += shift_pattern
139
140 assert_equal(res_patterns, cmp_patterns)
