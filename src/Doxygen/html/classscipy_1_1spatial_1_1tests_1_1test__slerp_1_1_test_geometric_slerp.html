278 # geometric_slerp() should handle scenarios with

288 # normalize to max of 1
289 forward_t_vals /= forward_t_vals.max()
290 reverse_t_vals = np.flipud(forward_t_vals)
291 shuffled_indices = np.arange(num_t_vals)
292 np.random.shuffle(shuffled_indices)
293 scramble_t_vals = forward_t_vals.copy()[shuffled_indices]
294
295 forward_results = geometric_slerp(start=start,
296 end=end,
297 t=forward_t_vals)
298 reverse_results = geometric_slerp(start=start,
299 end=end,
300 t=reverse_t_vals)

302 end=end,
303 t=scramble_t_vals)
