243 def disc_c2(x):
244 n, s = x.shape
245 xij = x
246 disc1 = np.sum(np.prod((1
247 + 1/2*np.abs(xij-0.5)
248 - 1/2*np.abs(xij-0.5)**2), axis=1))
249 xij = x[None, :, :]
250 xkj = x[:, None, :]
251 disc2 = np.sum(np.sum(np.prod(1
252 + 1/2*np.abs(xij - 0.5)
253 + 1/2*np.abs(xkj - 0.5)
254 - 1/2*np.abs(xij - xkj), axis=2),
255 axis=0))
256 return (13/12)**s - 2/n * disc1 + 1/n**2*disc2
257
258 def disc_wd(x):
259 n, s = x.shape
260 xij = x[None, :, :]
261 xkj = x[:, None, :]
262 disc = np.sum(np.sum(np.prod(3/2
263 - np.abs(xij - xkj)
264 + np.abs(xij - xkj)**2, axis=2),
265 axis=0))
266 return -(4/3)**s + 1/n**2 * disc
267
268 def disc_md(x):
269 n, s = x.shape
270 xij = x
271 disc1 = np.sum(np.prod((5/3
272 - 1/4*np.abs(xij-0.5)
273 - 1/4*np.abs(xij-0.5)**2), axis=1))
274 xij = x[None, :, :]
275 xkj = x[:, None, :]
276 disc2 = np.sum(np.sum(np.prod(15/8
277 - 1/4*np.abs(xij - 0.5)
278 - 1/4*np.abs(xkj - 0.5)
279 - 3/4*np.abs(xij - xkj)
280 + 1/2*np.abs(xij - xkj)**2,
281 axis=2), axis=0))
282 return (19/12)**s - 2/n * disc1 + 1/n**2*disc2

284 def disc_star_l2(x):
285 n, s = x.shape
286 return np.sqrt(

289 + np.sum([
290 np.prod(1 - np.maximum(x[k, :], x[j, :]))
291 for k in range(n) for j in range(n)
292 ]) / n ** 2
293 )
