461def _cubicmin(a, fa, fpa, b, fb, c, fc):
462 """
463 Finds the minimizer for a cubic polynomial that goes through the
464 points (a,fa), (b,fb), and (c,fc) with derivative at a of fpa.
465
466 If no minimizer can be found, return None.
467
468 """
469 # f(x) = A *(x-a)^3 + B*(x-a)^2 + C*(x-a) + D
470
471 with np.errstate(divide='raise', over='raise', invalid='raise'):
472 try:
473 C = fpa
474 db = b - a
475 dc = c - a
476 denom = (db * dc) ** 2 * (db - dc)
477 d1 = np.empty((2, 2))
478 d1[0, 0] = dc ** 2
479 d1[0, 1] = -db ** 2
480 d1[1, 0] = -dc ** 3
481 d1[1, 1] = db ** 3
482 [A, B] = np.dot(d1, np.asarray([fb - fa - C * db,
483 fc - fa - C * dc]).flatten())
484 A /= denom
485 B /= denom
486 radical = B * B - 3 * A * C
487 xmin = a + (-B + np.sqrt(radical)) / (3 * A)
488 except ArithmeticError:
489 return None
490 if not np.isfinite(xmin):
491 return None
492 return xmin

702 alpha1**2 * (phi_a0 - phi0 - derphi0*alpha0)
