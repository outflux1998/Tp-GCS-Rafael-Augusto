613def test_kwargs_in_init():
614 # Make sure the changed_only=True mode is OK when an argument is passed as
615 # kwargs.
616 # Non-regression test for
617 # https://github.com/scikit-learn/scikit-learn/issues/17206
618
619 class WithKWargs(BaseEstimator):
620 # Estimator with a kwargs argument. These need to hack around
621 # set_params and get_params. Here we mimic what LightGBM does.
622 def __init__(self, a="willchange", b="unchanged", **kwargs):
623 self.a = a
624 self.b = b
625 self._other_params = {}
626 self.set_params(**kwargs)
627
628 def get_params(self, deep=True):
629 params = super().get_params(deep=deep)
630 params.update(self._other_params)
631 return params
632
633 def set_params(self, **params):
634 for key, value in params.items():
635 setattr(self, key, value)
636 self._other_params[key] = value
637 return self
638
639 est = WithKWargs(a="something", c="abcd", d=None)
640
641 expected = "WithKWargs(a='something', c='abcd', d=None)"
642 assert expected == est.__repr__()
643
644 with config_context(print_changed_only=False):
645 expected = "WithKWargs(a='something', b='unchanged', c='abcd', d=None)"
646 assert expected == est.__repr__()
