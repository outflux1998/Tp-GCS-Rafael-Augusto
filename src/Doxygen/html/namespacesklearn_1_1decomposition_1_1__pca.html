Compute the log-likelihood of a rank ``rank`` dataset.

The dataset is assumed to be embedded in gaussian noise of shape(n,
dimf) having spectrum ``spectrum``. This implements the method of
T. P. Minka.

62 n_features = spectrum.shape[0]
63 if not 1 <= rank < n_features:
64 raise ValueError("the tested rank should be in [1, n_features - 1]")
65
66 eps = 1e-15
67
68 if spectrum[rank - 1] < eps:
69 # When the tested rank is associated with a small eigenvalue, there's
70 # no point in computing the log-likelihood: it's going to be very
71 # small and won't be the max anyway. Also, it can lead to numerical
72 # issues below when computing pa, in particular in log((spectrum[i] -
73 # spectrum[j]) because this will take the log of something very small.
74 return -np.inf
75
76 pu = -rank * log(2.0)
77 for i in range(1, rank + 1):
78 pu += (
79 gammaln((n_features - i + 1) / 2.0)
80 - log(np.pi) * (n_features - i + 1) / 2.0
81 )
82
83 pl = np.sum(np.log(spectrum[:rank]))
84 pl = -pl * n_samples / 2.0
85
86 v = max(eps, np.sum(spectrum[rank:]) / (n_features - rank))
87 pv = -np.log(v) * n_samples * (n_features - rank) / 2.0
88
89 m = n_features * rank - rank * (rank + 1.0) / 2.0
90 pp = log(2.0 * np.pi) * (m + rank) / 2.0
91
92 pa = 0.0
93 spectrum_ = spectrum.copy()
94 spectrum_[rank:n_features] = v
95 for i in range(rank):
96 for j in range(i + 1, len(spectrum)):
97 pa += log(
98 (spectrum[i] - spectrum[j]) * (1.0 / spectrum_[j] - 1.0 / spectrum_[i])
99 ) + log(n_samples)
100
101 ll = pu + pl + pv + pp - pa / 2.0 - rank * log(n_samples) / 2.0
102
103 return ll
