233 def test_convolve_method_large_input(self):
234 # This is really a test that convolving two large integers goes to the
235 # direct method even if they're in the fft method.
236 for n in [10, 20, 50, 51, 52, 53, 54, 60, 62]:
237 z = np.array([2**n], dtype=np.int64)
238 fft = convolve(z, z, method='fft')
239 direct = convolve(z, z, method='direct')
240
241 # this is the case when integer precision gets to us
242 # issue #6076 has more detail, hopefully more tests after resolved
243 if n < 50:
244 assert_equal(fft, direct)
245 assert_equal(fft, 2**(2*n))
246 assert_equal(direct, 2**(2*n))
