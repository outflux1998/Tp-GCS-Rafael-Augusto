920def test_dict_learning_numerical_consistency(method):
921 # verify numerically consistent among np.float32 and np.float64
922 rtol = 1e-6
923 n_components = 4
924 alpha = 2
925
926 U_64, V_64, _ = dict_learning(

939 )
940
941 # Optimal solution (U*, V*) is not unique.
942 # If (U*, V*) is optimal solution, (-U*,-V*) is also optimal,
943 # and (column permutated U*, row permutated V*) are also optional
944 # as long as holding UV.
945 # So here UV, ||U||_1,1 and sum(||V_k||_2^2) are verified
946 # instead of comparing directly U and V.
947 assert_allclose(np.matmul(U_64, V_64), np.matmul(U_32, V_32), rtol=rtol)

949 assert_allclose(np.sum(V_64**2), np.sum(V_32**2), rtol=rtol)
950 # verify an obtained solution is not degenerate
951 assert np.mean(U_64 != 0.0) > 0.05
952 assert np.count_nonzero(U_64 != 0.0) == np.count_nonzero(U_32 != 0.0)

67 def ricker_function(resolution, center, width):
68 """Discrete sub-sampled Ricker (Mexican hat) wavelet"""
69 x = np.linspace(0, resolution - 1, resolution)
70 x = (
71 (2 / (np.sqrt(3 * width) * np.pi**0.25))
72 * (1 - (x - center) ** 2 / width**2)
73 * np.exp(-((x - center) ** 2) / (2 * width**2))
74 )
75 return x
76
77 def ricker_matrix(width, resolution, n_components):
78 """Dictionary of Ricker (Mexican hat) wavelets"""
79 centers = np.linspace(0, resolution - 1, n_components)
80 D = np.empty((n_components, resolution))
81 for i, center in enumerate(centers):
82 D[i] = ricker_function(resolution, center, width)
83 D /= np.sqrt(np.sum(D**2, axis=1))[:, np.newaxis]
84 return D
