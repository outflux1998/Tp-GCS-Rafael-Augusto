Test that the solution to the discrete approximation of a continuous
system actually approximates the solution to the continuous system.
This is an indirect test of the correctness of the implementation

276 def test_discrete_approx(self):
277 """
278 Test that the solution to the discrete approximation of a continuous
279 system actually approximates the solution to the continuous system.
280 This is an indirect test of the correctness of the implementation
281 of cont2discrete.
282 """
283
284 def u(t):
285 return np.sin(2.5 * t)
286
287 a = np.array([[-0.01]])
288 b = np.array([[1.0]])
289 c = np.array([[1.0]])
290 d = np.array([[0.2]])
291 x0 = 1.0
292
293 t = np.linspace(0, 10.0, 101)
294 dt = t[1] - t[0]
295 u1 = u(t)
296
297 # Use lsim2 to compute the solution to the continuous system.
298 t, yout, xout = lsim2((a, b, c, d), T=t, U=u1, X0=x0,
299 rtol=1e-9, atol=1e-11)
300
301 # Convert the continuous system to a discrete approximation.
302 dsys = c2d((a, b, c, d), dt, method='bilinear')
303
304 # Use dlsim with the pairwise averaged input to compute the output
305 # of the discrete system.
306 u2 = 0.5 * (u1[:-1] + u1[1:])
307 t2 = t[:-1]
308 td2, yd2, xd2 = dlsim(dsys, u=u2.reshape(-1, 1), t=t2, x0=x0)
309
310 # ymid is the average of consecutive terms of the "exact" output
311 # computed by lsim2. This is what the discrete approximation
312 # actually approximates.
313 ymid = 0.5 * (yout[:-1] + yout[1:])
314
315 assert_allclose(yd2.ravel(), ymid, rtol=1e-4)

234 def test_gbt_with_sio_tf_and_zpk(self):
235 """Test method='gbt' with alpha=0.25 for tf and zpk cases."""
236 # State space coefficients for the continuous SIO system.
237 A = -1.0
238 B = 1.0
239 C = 1.0
240 D = 0.5
241
242 # The continuous transfer function coefficients.
243 cnum, cden = ss2tf(A, B, C, D)
244
245 # Continuous zpk representation
246 cz, cp, ck = ss2zpk(A, B, C, D)
247
248 h = 1.0
249 alpha = 0.25
250
251 # Explicit formulas, in the scalar case.
252 Ad = (1 + (1 - alpha) * h * A) / (1 - alpha * h * A)
253 Bd = h * B / (1 - alpha * h * A)
254 Cd = C / (1 - alpha * h * A)
255 Dd = D + alpha * C * Bd
256
257 # Convert the explicit solution to tf
258 dnum, dden = ss2tf(Ad, Bd, Cd, Dd)
259
260 # Compute the discrete tf using cont2discrete.
261 c2dnum, c2dden, dt = c2d((cnum, cden), h, method='gbt', alpha=alpha)
262
263 assert_allclose(dnum, c2dnum)
264 assert_allclose(dden, c2dden)
265
266 # Convert explicit solution to zpk.
267 dz, dp, dk = ss2zpk(Ad, Bd, Cd, Dd)
268
269 # Compute the discrete zpk using cont2discrete.
270 c2dz, c2dp, c2dk, dt = c2d((cz, cp, ck), h, method='gbt', alpha=alpha)
271
272 assert_allclose(dz, c2dz)
273 assert_allclose(dp, c2dp)
