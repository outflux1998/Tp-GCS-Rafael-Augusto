244def check_discrete_chisquare(distfn, arg, rvs, alpha, msg):
245 """Perform chisquare test for random sample of a discrete distribution
246
247 Parameters
248 ----------
249 distname : string
250 name of distribution function
251 arg : sequence
252 parameters of distribution
253 alpha : float
254 significance level, threshold for p-value
255
256 Returns
257 -------
258 result : bool
259 0 if test passes, 1 if test fails
260
261 """
262 wsupp = 0.05
263
264 # construct intervals with minimum mass `wsupp`.
265 # intervals are left-half-open as in a cdf difference
266 _a, _b = distfn.support(*arg)
267 lo = int(max(_a, -1000))
268 high = int(min(_b, 1000)) + 1
269 distsupport = range(lo, high)
270 last = 0
271 distsupp = [lo]
272 distmass = []
273 for ii in distsupport:
274 current = distfn.cdf(ii, *arg)
275 if current - last >= wsupp - 1e-14:
276 distsupp.append(ii)
277 distmass.append(current - last)
278 last = current
279 if current > (1 - wsupp):
280 break
281 if distsupp[-1] < _b:
282 distsupp.append(_b)
283 distmass.append(1 - last)
284 distsupp = np.array(distsupp)
285 distmass = np.array(distmass)
286
287 # convert intervals to right-half-open as required by histogram
288 histsupp = distsupp + 1e-8
289 histsupp[0] = _a
290
291 # find sample frequencies and perform chisquare test
292 freq, hsupp = np.histogram(rvs, histsupp)
293 chis, pval = stats.chisquare(np.array(freq), len(rvs)*distmass)
294
295 npt.assert_(pval > alpha,
296 'chisquare - test for %s at arg = %s with pval = %s' %
