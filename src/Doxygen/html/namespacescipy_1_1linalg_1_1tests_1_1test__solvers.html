294 def _test_factory(case, dec):
295 """Checks if 0 = XA + A'X - XB(R)^{-1} B'X + Q is true"""
296 a, b, q, r, knownfailure = case
297 if knownfailure:
298 pytest.xfail(reason=knownfailure)
299
300 x = solve_continuous_are(a, b, q, r)
301 res = x.dot(a) + a.conj().T.dot(x) + q
302 out_fact = x.dot(b)
303 res -= out_fact.dot(solve(np.atleast_2d(r), out_fact.conj().T))
304 assert_array_almost_equal(res, np.zeros_like(res), decimal=dec)
