366 return 0 if k < 0 or k > n else comb(int(n), int(k), exact=True)

369 n_samples = len(samples)
370 entropy = 0.0
371
372 for count in np.bincount(samples):
373 p = 1.0 * count / n_samples
374 if p > 0:
375 entropy -= p * np.log2(p)
376
377 return entropy

383 features.pop(X_m)
384 values = [np.unique(X[:, i]) for i in range(n_features)]
385

387
388 for k in range(n_features):
389 # Weight of each B of size k
390 coef = 1.0 / (binomial(k, n_features) * (n_features - k))
391

393 for B in combinations(features, k):
394 # For all values B=b
