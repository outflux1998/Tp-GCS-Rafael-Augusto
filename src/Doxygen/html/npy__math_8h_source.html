393/*
394 * C99 specifies that complex numbers have the same representation as
395 * an array of two elements, where the first element is the real part
396 * and the second element is the imaginary part.
397 */
398#define __NPY_CPACK_IMP(x, y, type, ctype) \
399 union { \
400 ctype z; \
401 type a[2]; \
402 } z1; \
403 \
404 z1.a[0] = (x); \
405 z1.a[1] = (y); \
406 \
407 return z1.z;
408
409static NPY_INLINE npy_cdouble npy_cpack(double x, double y)
410{
411 __NPY_CPACK_IMP(x, y, double, npy_cdouble);
412}
413
414static NPY_INLINE npy_cfloat npy_cpackf(float x, float y)
415{
416 __NPY_CPACK_IMP(x, y, float, npy_cfloat);
417}
418
419static NPY_INLINE npy_clongdouble npy_cpackl(npy_longdouble x, npy_longdouble y)
420{

422}
423#undef __NPY_CPACK_IMP

425/*
426 * Same remark as above, but in the other direction: extract first/second
427 * member of complex number, assuming a C99-compatible representation
428 *
429 * Those are defineds as static inline, and such as a reasonable compiler would
430 * most likely compile this to one or two instructions (on CISC at least)
431 */
432#define __NPY_CEXTRACT_IMP(z, index, type, ctype) \
433 union { \
434 ctype z; \
435 type a[2]; \
436 } __z_repr; \
437 __z_repr.z = z; \
438 \
439 return __z_repr.a[index];
440
441static NPY_INLINE double npy_creal(npy_cdouble z)
442{
443 __NPY_CEXTRACT_IMP(z, 0, double, npy_cdouble);
444}

447{

449}
