361 def test_large_chebyshev(self):
362 # The weights for Chebyshev points of the second kind have analytically
363 # solvable weights. Naive calculation of barycentric weights will fail
364 # for large N because of numerical underflow and overflow. We test
365 # correctness for large N against analytical Chebyshev weights.
366
367 # Without capacity scaling or permutation, n=800 fails,
368 # With just capacity scaling, n=1097 fails
369 # With both capacity scaling and random permutation, n=30000 succeeds
370 n = 800
371 j = np.arange(n + 1).astype(np.float64)
372 x = np.cos(j * np.pi / n)
373
374 # See page 506 of Berrut and Trefethen 2004 for this formula
375 w = (-1) ** j
376 w[0] *= 0.5
377 w[-1] *= 0.5
378
379 P = BarycentricInterpolator(x)
380
381 # It's okay to have a constant scaling factor in the weights because it
382 # cancels out in the evaluation of the polynomial.
383 factor = P.wi[0]
384 assert_almost_equal(P.wi / (2 * factor), w)
385
