Detailed Description

Support code for building Python extensions on Windows.

    # NT stuff
    # 1. Make sure libpython<version>.a exists for gcc.  If not, build it.
    # 2. Force windows to use gcc (we're struggling with MSVC and g77 support)
    # 3. Force windows to use g77

356 major_version, minor_version = tuple(sys.version_info[:2])

358 # patterns for the file name of the library itself
359 patterns = ['libpython%d%d.a',
360 'libpython%d%d.dll.a',
361 'libpython%d.%d.dll.a']
362
363 # directory trees that may contain the library
364 stems = [sys.prefix]
365 if hasattr(sys, 'base_prefix') and sys.base_prefix != sys.prefix:
366 stems.append(sys.base_prefix)
367 elif hasattr(sys, 'real_prefix') and sys.real_prefix != sys.prefix:
368 stems.append(sys.real_prefix)
369
370 # possible subdirectories within those trees where it is placed
371 sub_dirs = ['libs', 'lib']
372
373 # generate a list of candidate locations
374 candidates = []
375 for pat in patterns:
376 filename = pat % (major_version, minor_version)
377 for stem_dir in stems:
378 for folder in sub_dirs:
379 candidates.append(os.path.join(stem_dir, folder, filename))
380
381 # test the filesystem to see if we can find any of these
382 for fullname in candidates:
383 if os.path.isfile(fullname):
384 # already exists, in location given
385 return (True, fullname)
386
387 # needs to be built, preferred location given first
388 return (False, candidates[0])
