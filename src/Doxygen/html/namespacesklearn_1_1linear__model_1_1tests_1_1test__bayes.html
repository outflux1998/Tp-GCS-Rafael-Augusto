34def test_bayesian_ridge_score_values():
35 """Check value of score on toy example.
36
37 Compute log marginal likelihood with equation (36) in Sparse Bayesian
38 Learning and the Relevance Vector Machine (Tipping, 2001):
39
40 - 0.5 * (log |Id/alpha + X.X^T/lambda| +
41 y^T.(Id/alpha + X.X^T/lambda).y + n * log(2 * pi))
42 + lambda_1 * log(lambda) - lambda_2 * lambda
43 + alpha_1 * log(alpha) - alpha_2 * alpha
44
45 and check equality with the score computed during training.
46 """
47
48 X, y = diabetes.data, diabetes.target
49 n_samples = X.shape[0]
50 # check with initial values of alpha and lambda (see code for the values)
51 eps = np.finfo(np.float64).eps
52 alpha_ = 1.0 / (np.var(y) + eps)
53 lambda_ = 1.0
54
55 # value of the parameters of the Gamma hyperpriors
56 alpha_1 = 0.1
57 alpha_2 = 0.1
58 lambda_1 = 0.1
59 lambda_2 = 0.1
60
61 # compute score using formula of docstring
62 score = lambda_1 * log(lambda_) - lambda_2 * lambda_
63 score += alpha_1 * log(alpha_) - alpha_2 * alpha_
64 M = 1.0 / alpha_ * np.eye(n_samples) + 1.0 / lambda_ * np.dot(X, X.T)
65 M_inv_dot_y = np.linalg.solve(M, y)
66 score += -0.5 * (
67 fast_logdet(M) + np.dot(y.T, M_inv_dot_y) + n_samples * log(2 * np.pi)
68 )
69
70 # compute score with BayesianRidge
71 clf = BayesianRidge(
72 alpha_1=alpha_1,
73 alpha_2=alpha_2,
74 lambda_1=lambda_1,
75 lambda_2=lambda_2,
76 n_iter=1,
77 fit_intercept=False,
78 compute_score=True,
79 )
80 clf.fit(X, y)
81
82 assert_almost_equal(clf.scores_[0], score, decimal=9)
