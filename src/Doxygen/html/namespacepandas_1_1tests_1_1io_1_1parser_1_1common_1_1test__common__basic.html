37def test_override_set_noconvert_columns():
38 # see gh-17351
39 #
40 # Usecols needs to be sorted in _set_noconvert_columns based
41 # on the test_usecols_with_parse_dates test from test_usecols.py
42 class MyTextFileReader(TextFileReader):
43 def __init__(self) -> None:
44 self._currow = 0
45 self.squeeze = False
46
47 class MyCParserWrapper(CParserWrapper):
48 def _set_noconvert_columns(self):
49 if self.usecols_dtype == "integer":
50 # self.usecols is a set, which is documented as unordered
51 # but in practice, a CPython set of integers is sorted.
52 # In other implementations this assumption does not hold.
53 # The following code simulates a different order, which
54 # before GH 17351 would cause the wrong columns to be
55 # converted via the parse_dates parameter
56 self.usecols = list(self.usecols)
57 self.usecols.reverse()
58 return CParserWrapper._set_noconvert_columns(self)
59
60 data = """a,b,c,d,e
610,1,20140101,0900,4
620,1,20140102,1000,4"""
63
64 parse_dates = [[1, 2]]
65 cols = {
66 "a": [0, 0],
67 "c_d": [Timestamp("2014-01-01 09:00:00"), Timestamp("2014-01-02 10:00:00")],
68 }
69 expected = DataFrame(cols, columns=["c_d", "a"])
70
71 parser = MyTextFileReader()
72 parser.options = {
73 "usecols": [0, 2, 3],
74 "parse_dates": parse_dates,
75 "delimiter": ",",
76 }
77 parser.engine = "c"
78 parser._engine = MyCParserWrapper(StringIO(data), **parser.options)
79
80 result = parser.read()
81 tm.assert_frame_equal(result, expected)
