33def _zpkbilinear(z, p, k, fs):
34 """Bilinear transformation to convert a filter from analog to digital."""
35
36 degree = _relative_degree(z, p)
37
38 fs2 = 2*fs
39
40 # Bilinear transform the poles and zeros
41 z_z = [(fs2 + z1) / (fs2 - z1) for z1 in z]
42 p_z = [(fs2 + p1) / (fs2 - p1) for p1 in p]
43
44 # Any zeros that were at infinity get moved to the Nyquist frequency
45 z_z.extend([-1] * degree)

50 k_z = k * numer / denom

55def _zpklp2lp(z, p, k, wo=1):
56 """Transform a lowpass filter to a different cutoff frequency."""
57
