395def _estimate_log_gaussian_prob(X, means, precisions_chol, covariance_type):
396 """Estimate the log Gaussian probability.
397
398 Parameters
399 ----------
400 X : array-like of shape (n_samples, n_features)
401
402 means : array-like of shape (n_components, n_features)
403
404 precisions_chol : array-like
405 Cholesky decompositions of the precision matrices.
406 'full' : shape of (n_components, n_features, n_features)
407 'tied' : shape of (n_features, n_features)
408 'diag' : shape of (n_components, n_features)
409 'spherical' : shape of (n_components,)
410
411 covariance_type : {'full', 'tied', 'diag', 'spherical'}
412
413 Returns
414 -------
415 log_prob : array, shape (n_samples, n_components)
416 """
417 n_samples, n_features = X.shape
418 n_components, _ = means.shape
419 # The determinant of the precision matrix from the Cholesky decomposition
420 # corresponds to the negative half of the determinant of the full precision
421 # matrix.
422 # In short: det(precision_chol) = - det(precision) / 2
423 log_det = _compute_log_det_cholesky(precisions_chol, covariance_type, n_features)
424
425 if covariance_type == "full":
426 log_prob = np.empty((n_samples, n_components))
427 for k, (mu, prec_chol) in enumerate(zip(means, precisions_chol)):
428 y = np.dot(X, prec_chol) - np.dot(mu, prec_chol)
429 log_prob[:, k] = np.sum(np.square(y), axis=1)
430
431 elif covariance_type == "tied":
432 log_prob = np.empty((n_samples, n_components))
433 for k, mu in enumerate(means):
434 y = np.dot(X, precisions_chol) - np.dot(mu, precisions_chol)
435 log_prob[:, k] = np.sum(np.square(y), axis=1)
436
437 elif covariance_type == "diag":
438 precisions = precisions_chol**2
439 log_prob = (
440 np.sum((means**2 * precisions), 1)
441 - 2.0 * np.dot(X, (means * precisions).T)
442 + np.dot(X**2, precisions.T)
443 )
444
445 elif covariance_type == "spherical":
446 precisions = precisions_chol**2
447 log_prob = (
448 np.sum(means**2, 1) * precisions
449 - 2 * np.dot(X, means.T * precisions)
450 + np.outer(row_norms(X, squared=True), precisions)
451 )
452 # Since we are using the precision of the Cholesky decomposition,
453 # `- 0.5 * log_det_precision` becomes `+ log_det_precision_chol`
454 return -0.5 * (n_features * np.log(2 * np.pi) + log_prob) + log_det
