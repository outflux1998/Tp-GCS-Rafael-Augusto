418def _beta_divergence_dense(X, W, H, beta):
419 """Compute the beta-divergence of X and W.H for dense array only.
420
421 Used as a reference for testing nmf._beta_divergence.
422 """
423 WH = np.dot(W, H)
424
425 if beta == 2:
426 return squared_norm(X - WH) / 2
427
428 WH_Xnonzero = WH[X != 0]
429 X_nonzero = X[X != 0]
430 np.maximum(WH_Xnonzero, 1e-9, out=WH_Xnonzero)
431
432 if beta == 1:
433 res = np.sum(X_nonzero * np.log(X_nonzero / WH_Xnonzero))
434 res += WH.sum() - X.sum()
435
436 elif beta == 0:
437 div = X_nonzero / WH_Xnonzero
438 res = np.sum(div) - X.size - np.sum(np.log(div))
439 else:
440 res = (X_nonzero**beta).sum()
441 res += (beta - 1) * (WH**beta).sum()
442 res -= beta * (X_nonzero * (WH_Xnonzero ** (beta - 1))).sum()
443 res /= beta * (beta - 1)
444
445 return res

702def test_nmf_decreasing(solver):
703 # test that the objective function is decreasing at each iteration
704 n_samples = 20
705 n_features = 15
706 n_components = 10
707 alpha = 0.1
708 l1_ratio = 0.5
709 tol = 0.0
710
711 # initialization
712 rng = np.random.mtrand.RandomState(42)
713 X = rng.randn(n_samples, n_features)
714 np.abs(X, X)
715 W0, H0 = nmf._initialize_nmf(X, n_components, init="random", random_state=42)
716
717 for beta_loss in (-1.2, 0, 0.2, 1.0, 2.0, 2.5):
718 if solver != "mu" and beta_loss != 2:
719 # not implemented
720 continue
721 W, H = W0.copy(), H0.copy()
722 previous_loss = None
723 for _ in range(30):
724 # one more iteration starting from the previous results
725 W, H, _ = non_negative_factorization(
726 X,
727 W,
728 H,
729 beta_loss=beta_loss,
730 init="custom",
731 n_components=n_components,
732 max_iter=1,
733 alpha_W=alpha,
734 solver=solver,
735 tol=tol,
736 l1_ratio=l1_ratio,
737 verbose=0,
738 random_state=0,
739 update_H=True,
740 )
741
742 loss = (
743 nmf._beta_divergence(X, W, H, beta_loss)
744 + alpha * l1_ratio * n_features * W.sum()
745 + alpha * l1_ratio * n_samples * H.sum()
746 + alpha * (1 - l1_ratio) * n_features * (W**2).sum()
747 + alpha * (1 - l1_ratio) * n_samples * (H**2).sum()
748 )
749 if previous_loss is not None:
750 assert previous_loss > loss
751 previous_loss = loss
