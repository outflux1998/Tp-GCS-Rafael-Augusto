joblib.test.test_hashing.test_hashes_stay_the_same_with_numpy_objects ( )
415def test_hashes_stay_the_same_with_numpy_objects():
416 # Note: joblib used to test numpy objects hashing by comparing the produced
417 # hash of an object with some hard-coded target value to guarantee that
418 # hashing remains the same across joblib versions. However, since numpy
419 # 1.20 and joblib 1.0, joblib relies on potentially unstable implementation
420 # details of numpy to hash np.dtype objects, which makes the stability of
421 # hash values across different environments hard to guarantee and to test.
422 # As a result, hashing stability across joblib versions becomes best-effort
423 # only, and we only test the consistency within a single environment by
424 # making sure:
425 # - the hash of two copies of the same objects is the same
426 # - hashing some object in two different python processes produces the same
427 # value. This should be viewed as a proxy for testing hash consistency
428 # through time between Python sessions (provided no change in the
429 # environment was done between sessions).
430
431 def create_objects_to_hash():
432 rng = np.random.RandomState(42)
433 # Being explicit about dtypes in order to avoid
434 # architecture-related differences. Also using 'f4' rather than
435 # 'f8' for float arrays because 'f8' arrays generated by
436 # rng.random.randn don't seem to be bit-identical on 32bit and
437 # 64bit machines.
438 to_hash_list = [
439 rng.randint(-1000, high=1000, size=50).astype('<i8'),
440 tuple(rng.randn(3).astype('<f4') for _ in range(5)),
441 [rng.randn(3).astype('<f4') for _ in range(5)],
442 {
443 -3333: rng.randn(3, 5).astype('<f4'),
444 0: [
445 rng.randint(10, size=20).astype('<i8'),
446 rng.randn(10).astype('<f4')
447 ]
448 },
449 # Non regression cases for
450 # https://github.com/joblib/joblib/issues/308
451 np.arange(100, dtype='<i8').reshape((10, 10)),
452 # Fortran contiguous array
453 np.asfortranarray(np.arange(100, dtype='<i8').reshape((10, 10))),
454 # Non contiguous array
455 np.arange(100, dtype='<i8').reshape((10, 10))[:, :2],
456 ]
457 return to_hash_list
458
459 # Create two lists containing copies of the same objects. joblib.hash
460 # should return the same hash for to_hash_list_one[i] and
461 # to_hash_list_two[i]
462 to_hash_list_one = create_objects_to_hash()
463 to_hash_list_two = create_objects_to_hash()
464
465 e1 = ProcessPoolExecutor(max_workers=1)
466 e2 = ProcessPoolExecutor(max_workers=1)
467
468 try:
469 for obj_1, obj_2 in zip(to_hash_list_one, to_hash_list_two):
470 # testing consistency of hashes across python processes
471 hash_1 = e1.submit(hash, obj_1).result()
472 hash_2 = e2.submit(hash, obj_1).result()
473 assert hash_1 == hash_2
474
475 # testing consistency when hashing two copies of the same objects.
476 hash_3 = e1.submit(hash, obj_2).result()
477 assert hash_1 == hash_3
478
479 finally:
480 e1.shutdown()
481 e2.shutdown()
