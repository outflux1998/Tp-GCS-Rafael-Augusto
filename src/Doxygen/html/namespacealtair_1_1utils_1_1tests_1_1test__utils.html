35def test_sanitize_dataframe():
36 # create a dataframe with various types
37 df = pd.DataFrame(
38 {
39 "s": list("abcde"),
40 "f": np.arange(5, dtype=float),
41 "i": np.arange(5, dtype=int),
42 "b": np.array([True, False, True, True, False]),
43 "d": pd.date_range("2012-01-01", periods=5, freq="H"),
44 "c": pd.Series(list("ababc"), dtype="category"),
45 "c2": pd.Series([1, "A", 2.5, "B", None], dtype="category"),
46 "o": pd.Series([np.array(i) for i in range(5)]),
47 "p": pd.date_range("2012-01-01", periods=5, freq="H").tz_localize("UTC"),
48 }
49 )
50
51 # add some nulls
52 df.iloc[0, df.columns.get_loc("s")] = None
53 df.iloc[0, df.columns.get_loc("f")] = np.nan
54 df.iloc[0, df.columns.get_loc("d")] = pd.NaT
55 df.iloc[0, df.columns.get_loc("o")] = np.array(np.nan)
56
57 # JSON serialize. This will fail on non-sanitized dataframes
58 print(df[["s", "c2"]])
59 df_clean = sanitize_dataframe(df)
60 print(df_clean[["s", "c2"]])
61 print(df_clean[["s", "c2"]].to_dict())
62 s = json.dumps(df_clean.to_dict(orient="records"))
63 print(s)
64
65 # Re-construct pandas dataframe
66 df2 = pd.read_json(s)
67
68 # Re-order the columns to match df
69 df2 = df2[df.columns]
70
71 # Re-apply original types
72 for col in df:
73 if str(df[col].dtype).startswith("datetime"):
74 # astype(datetime) introduces time-zone issues:
75 # to_datetime() does not.
76 utc = isinstance(df[col].dtype, pd.core.dtypes.dtypes.DatetimeTZDtype)
77 df2[col] = pd.to_datetime(df2[col], utc=utc)
78 else:
79 df2[col] = df2[col].astype(df[col].dtype)
80
81 # pandas doesn't properly recognize np.array(np.nan), so change it here
82 df.iloc[0, df.columns.get_loc("o")] = np.nan
83 assert df.equals(df2)
