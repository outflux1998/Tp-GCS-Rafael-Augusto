564def _determine_karray(summed_widths, equal_heights, total_width, max_height):
565 n = len(equal_heights)
566 eq_rs, eq_as = np.asarray(equal_heights).T
567 sm_rs, sm_as = np.asarray(summed_widths).T
568 A = np.zeros((n + 1, n + 1))
569 B = np.zeros(n + 1)
570 np.fill_diagonal(A[:n, :n], eq_rs)
571 A[:n, -1] = -1
572 A[-1, :-1] = sm_rs
573 B[:n] = -eq_as
574 B[-1] = total_width - sum(sm_as)
575 # A @ K = B: This solves for {k_0, ..., k_{N-1}, H} so that
576 # eq_r_i * k_i + eq_a_i = H for all i: all axes have the same height
577 # sum(sm_r_i * k_i + sm_a_i) = total_summed_width: fixed total width
578 # (foo_r_i * k_i + foo_a_i will end up being the size of foo.)
579 karray_and_height = np.linalg.solve(A, B)
580 karray = karray_and_height[:-1]
581 height = karray_and_height[-1]
582 if height > max_height: # Additionally, upper-bound the height.
583 karray = (max_height - eq_as) / eq_rs
584 return karray
