452 def compile(self, ttFont):
453 # First make sure that all the data lines up properly. Formats 1 and 3
454 # must have all its data lined up consecutively. If not this will fail.
455 for curLoc, nxtLoc in zip(self.locations, self.locations[1:]):
456 assert curLoc[1] == nxtLoc[0], "Data must be consecutive in indexSubTable offset formats"
457
458 glyphIds = list(map(ttFont.getGlyphID, self.names))
459 # Make sure that all ids are sorted strictly increasing.
460 assert all(glyphIds[i] < glyphIds[i+1] for i in range(len(glyphIds)-1))
461
462 # Run a simple algorithm to add skip glyphs to the data locations at
463 # the places where an id is not present.
464 idQueue = deque(glyphIds)
465 locQueue = deque(self.locations)
466 allGlyphIds = list(range(self.firstGlyphIndex, self.lastGlyphIndex+1))
467 allLocations = []
468 for curId in allGlyphIds:
469 if curId != idQueue[0]:
470 allLocations.append((locQueue[0][0], locQueue[0][0]))
471 else:
472 idQueue.popleft()
473 allLocations.append(locQueue.popleft())
474
475 # Now that all the locations are collected, pack them appropriately into
476 # offsets. This is the form where offset[i] is the location and
477 # offset[i+1]-offset[i] is the size of the data location.
478 offsets = list(allLocations[0]) + [loc[1] for loc in allLocations[1:]]
479 # Image data offset must be less than or equal to the minimum of locations.
480 # This offset may change the value for round tripping but is safer and
481 # allows imageDataOffset to not be required to be in the XML version.
482 self.imageDataOffset = min(offsets)
483 offsetArray = [offset - self.imageDataOffset for offset in offsets]
484
485 dataList = [EblcIndexSubTable.compile(self, ttFont)]
486 dataList += [struct.pack(dataFormat, offsetValue) for offsetValue in offsetArray]
487 # Take care of any padding issues. Only occurs in format 3.
488 if offsetDataSize * len(offsetArray) % 4 != 0:
489 dataList.append(struct.pack(dataFormat, 0))
490 return bytesjoin(dataList)
