474 # Recursion for Hermite functions
475 def hermite_recursion(n, nodes):
476 H = np.zeros((n, nodes.size))
477 H[0,:] = np.pi**(-0.25) * np.exp(-0.5*nodes**2)
478 if n > 1:
479 H[1,:] = sqrt(2.0) * nodes * H[0,:]
480 for k in range(2, n):
481 H[k,:] = sqrt(2.0/k) * nodes * H[k-1,:] - sqrt((k-1.0)/k) * H[k-2,:]
482 return H
483
484 # This tests only the nodes
485 def test(N, rtol=1e-15, atol=1e-14):
486 x, w = orth._roots_hermite_asy(N)
487 H = hermite_recursion(N+1, x)
488 assert_allclose(H[-1,:], np.zeros(N), rtol, atol)
489 assert_allclose(sum(w), sqrt(np.pi), rtol, atol)
