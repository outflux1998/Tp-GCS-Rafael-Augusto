385def remove_hints(self):
386 cff = self.cff
387 for fontname in cff.keys():
388 font = cff[fontname]
389 cs = font.CharStrings
390 # This can be tricky, but doesn't have to. What we do is:
391 #
392 # - Run all used glyph charstrings and recurse into subroutines,
393 # - For each charstring (including subroutines), if it has any
394 # of the hint stem operators, we mark it as such.
395 # Upon returning, for each charstring we note all the
396 # subroutine calls it makes that (recursively) contain a stem,
397 # - Dropping hinting then consists of the following two ops:
398 # * Drop the piece of the program in each charstring before the
399 # last call to a stem op or a stem-calling subroutine,
400 # * Drop all hintmask operations.
401 # - It's trickier... A hintmask right after hints and a few numbers
402 # will act as an implicit vstemhm. As such, we track whether
403 # we have seen any non-hint operators so far and do the right
404 # thing, recursively... Good luck understanding that :(
405 css = set()
406 for g in font.charset:
407 c, _ = cs.getItemAndSelector(g)
408 c.decompile()
409 subrs = getattr(c.private, "Subrs", [])
410 decompiler = _DehintingT2Decompiler(css, subrs, c.globalSubrs,
411 c.private.nominalWidthX,
412 c.private.defaultWidthX,
413 c.private)
414 decompiler.execute(c)
415 c.width = decompiler.width
416 for charstring in css:
417 charstring.drop_hints()
418 del css
419
420 # Drop font-wide hinting values
421 all_privs = []
422 if hasattr(font, 'FDArray'):
423 all_privs.extend(fd.Private for fd in font.FDArray)
424 else:
425 all_privs.append(font.Private)
426 for priv in all_privs:
427 for k in ['BlueValues', 'OtherBlues',
428 'FamilyBlues', 'FamilyOtherBlues',
429 'BlueScale', 'BlueShift', 'BlueFuzz',
430 'StemSnapH', 'StemSnapV', 'StdHW', 'StdVW',
431 'ForceBold', 'LanguageGroup', 'ExpansionFactor']:
432 if hasattr(priv, k):
433 setattr(priv, k, None)
