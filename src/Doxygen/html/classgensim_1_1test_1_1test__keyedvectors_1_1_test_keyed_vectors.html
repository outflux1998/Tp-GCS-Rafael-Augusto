117 wordnet_syn = [
118 'good', 'goodness', 'commodity', 'trade_good', 'full', 'estimable', 'honorable',
119 'respectable', 'beneficial', 'just', 'upright', 'adept', 'expert', 'practiced', 'proficient',

123 ] # synonyms for "good" as per wordnet
124 cos_sim = []

126 if wordnet_syn[i] in self.vectors.vocab:
127 cos_sim.append(self.vectors.similarity("good", wordnet_syn[i]))
128 cos_sim = sorted(cos_sim, reverse=True) # cosine_similarity of "good" with wordnet_syn in decreasing order
129 # computing relative_cosine_similarity of two similar words
130 rcs_wordnet = self.vectors.similarity("good", "nice") / sum(cos_sim[i] for i in range(10))
131 rcs = self.vectors.relative_cosine_similarity("good", "nice", 10)
132 self.assertTrue(rcs_wordnet >= rcs)
