1/*
2 * This header exports the new experimental DType API as proposed in
3 * NEPs 41 to 43. For background, please check these NEPs. Otherwise,
4 * this header also serves as documentation for the time being.
5 *
6 * Please do not hesitate to contact @seberg with questions. This is
7 * developed together with https://github.com/seberg/experimental_user_dtypes
8 * and those interested in experimenting are encouraged to contribute there.
9 *
10 * To use the functions defined in the header, call::
11 *
12 * if (import_experimental_dtype_api(version) < 0) {
13 * return NULL;
14 * }
15 *
16 * in your module init. (A version mismatch will be reported, just update
17 * to the correct one, this will alert you of possible changes.)
18 *
19 * The following lists the main symbols currently exported. Please do not
20 * hesitate to ask for help or clarification:
21 *
22 * - PyUFunc_AddLoopFromSpec:
23 *
24 * Register a new loop for a ufunc. This uses the `PyArrayMethod_Spec`
25 * which must be filled in (see in-line comments).
26 *
27 * - PyUFunc_AddWrappingLoop:
28 *
29 * Register a new loop which reuses an existing one, but modifies the
30 * result dtypes. Please search the internal NumPy docs for more info
31 * at this point. (Used for physical units dtype.)
32 *
33 * - PyUFunc_AddPromoter:
34 *
35 * Register a new promoter for a ufunc. A promoter is a function stored
36 * in a PyCapsule (see in-line comments). It is passed the operation and
37 * requested DType signatures and can mutate it to attempt a new search
38 * for a matching loop/promoter.
39 * I.e. for Numba a promoter could even add the desired loop.
40 *
41 * - PyArrayInitDTypeMeta_FromSpec:
42 *
43 * Initialize a new DType. It must currently be a static Python C type
44 * that is declared as `PyArray_DTypeMeta` and not `PyTypeObject`.
45 * Further, it must subclass `np.dtype` and set its type to
46 * `PyArrayDTypeMeta_Type` (before calling `PyType_Read()`).
47 *
48 * - PyArray_CommonDType:
49 *
50 * Find the common-dtype ("promotion") for two DType classes. Similar
51 * to `np.result_type`, but works on the classes and not instances.
52 *
53 * - PyArray_PromoteDTypeSequence:
54 *
55 * Same as CommonDType, but works with an arbitrary number of DTypes.
56 * This function is smarter and can often return successful and unambiguous
57 * results when `common_dtype(common_dtype(dt1, dt2), dt3)` would
58 * depend on the operation order or fail. Nevertheless, DTypes should
59 * aim to ensure that their common-dtype implementation is associative
60 * and commutative! (Mainly, unsigned and signed integers are not.)
61 *
62 * For guaranteed consistent results DTypes must implement common-Dtype
63 * "transitively". If A promotes B and B promotes C, than A must generally
64 * also promote C; where "promotes" means implements the promotion.
65 * (There are some exceptions for abstract DTypes)
66 *
67 * - PyArray_GetDefaultDescr:
68 *
69 * Given a DType class, returns the default instance (descriptor).
70 * This is an inline function checking for `singleton` first and only
71 * calls the `default_descr` function if necessary.
72 *
73 * - PyArray_DoubleDType, etc.:
74 *
75 * Aliases to the DType classes for the builtin NumPy DTypes.
76 *
77 * WARNING
78 * =======
79 *
80 * By using this header, you understand that this is a fully experimental
81 * exposure. Details are expected to change, and some options may have no
82 * effect. (Please contact @seberg if you have questions!)
83 * If the exposure stops working, please file a bug report with NumPy.
84 * Further, a DType created using this API/header should still be expected
85 * to be incompatible with some functionality inside and outside of NumPy.
86 * In this case crashes must be expected. Please report any such problems
87 * so that they can be fixed before final exposure.
88 * Furthermore, expect missing checks for programming errors which the final
89 * API is expected to have.
90 *
91 * Symbols with a leading underscore are likely to not be included in the
92 * first public version, if these are central to your use-case, please let
93 * us know, so that we can reconsider.
94 *
95 * "Array-like" consumer API not yet under considerations
96 * ======================================================
97 *
98 * The new DType API is designed in a way to make it potentially useful for
99 * alternative "array-like" implementations. This will require careful
100 * exposure of details and functions and is not part of this experimental API.
101 *
102 * Brief (incompatibility) changelog
103 * =================================
104 *
105 * 2. None (only additions).
106 * 3. New `npy_intp *view_offset` argument for `resolve_descriptors`.
107 * This replaces the `NPY_CAST_IS_VIEW` flag. It can be set to 0 if the
108 * operation is a view, and is pre-initialized to `NPY_MIN_INTP` indicating
109 * that the operation is not a view.
110 */

449/*
450 * ********************************
451 * Initialization
452 * ********************************
453 *
454 * Import the experimental API, the version must match the one defined in
455 * the header to ensure changes are taken into account. NumPy will further
456 * runtime-check this.
457 * You must call this function to use the symbols defined in this file.
458 */
459#if !defined(NO_IMPORT) && !defined(NO_IMPORT_ARRAY)
460
461#define __EXPERIMENTAL_DTYPE_VERSION 5
462
463static int
464import_experimental_dtype_api(int version)
465{
466 if (version != __EXPERIMENTAL_DTYPE_VERSION) {
467 PyErr_Format(PyExc_RuntimeError,
468 "DType API version %d did not match header version %d. Please "
469 "update the import statement and check for API changes.",
470 version, __EXPERIMENTAL_DTYPE_VERSION);
471 return -1;
472 }
473 if (__experimental_dtype_api_table != __uninitialized_table) {
474 /* already imported. */
475 return 0;
476 }
477
478 PyObject *multiarray = PyImport_ImportModule("numpy.core._multiarray_umath");
479 if (multiarray == NULL) {
480 return -1;
481 }
482
483 PyObject *api = PyObject_CallMethod(multiarray,
484 "_get_experimental_dtype_api", "i", version);
485 Py_DECREF(multiarray);
486 if (api == NULL) {
487 return -1;
488 }
489 __experimental_dtype_api_table = (void **)PyCapsule_GetPointer(api,
490 "experimental_dtype_api_table");
491 Py_DECREF(api);
492
493 if (__experimental_dtype_api_table == NULL) {
494 __experimental_dtype_api_table = __uninitialized_table;
495 return -1;
496 }
497 return 0;
498}
