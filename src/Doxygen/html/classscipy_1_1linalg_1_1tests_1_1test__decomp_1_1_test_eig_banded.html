401 # hermitian band matrix
402 self.herm_mat = (diag(full(N, -1.0))
403 + 1j*diag(full(N-1, 1.0), -1)
404 - 1j*diag(full(N-1, 1.0), 1)
405 + diag(full(N-2, -2.0), -2)
406 + diag(full(N-2, -2.0), 2))
407
408 # general real band matrix
409 self.real_mat = (diag(full(N, 1.0))
410 + diag(full(N-1, -1.0), -1) + diag(full(N-1, -3.0), 1)
411 + diag(full(N-2, 2.0), -2) + diag(full(N-2, -2.0), 2))
412
413 # general complex band matrix
414 self.comp_mat = (1j*diag(full(N, 1.0))
415 + diag(full(N-1, -1.0), -1)
416 + 1j*diag(full(N-1, -3.0), 1)
417 + diag(full(N-2, 2.0), -2)
418 + diag(full(N-2, -2.0), 2))
