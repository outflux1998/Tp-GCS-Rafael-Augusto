758 # Simple test, designed to excersize most specialized code paths,
759 # note the +0.5 for floats. This makes sure we use a float value

761 arr = (np.arange(7) + 0.5).astype(dtype)
762 scalar = np.array(2, dtype=dtype)
763
764 # contig -> scalar:
765 res = np.einsum('i->', arr)
766 assert res == arr.sum()
767 # contig, contig -> contig:
768 res = np.einsum('i,i->i', arr, arr)
769 assert_array_equal(res, arr * arr)
770 # noncontig, noncontig -> contig:
771 res = np.einsum('i,i->i', arr.repeat(2)[::2], arr.repeat(2)[::2])
772 assert_array_equal(res, arr * arr)
773 # contig + contig -> scalar
774 assert np.einsum('i,i->', arr, arr) == (arr * arr).sum()
775 # contig + scalar -> contig (with out)
776 out = np.ones(7, dtype=dtype)
777 res = np.einsum('i,->i', arr, dtype.type(2), out=out)
778 assert_array_equal(res, arr * dtype.type(2))
779 # scalar + contig -> contig (with out)
780 res = np.einsum(',i->i', scalar, arr)
781 assert_array_equal(res, arr * dtype.type(2))
782 # scalar + contig -> scalar
783 res = np.einsum(',i->', scalar, arr)
784 # Use einsum to compare to not have difference due to sum round-offs:
785 assert res == np.einsum('i->', scalar * arr)
