        We first re-order the master coordinate values.
        For a moveto to lineto, the args are now arranged as::

            [ [master_0 x,y], [master_1 x,y], [master_2 x,y] ]

        We re-arrange this to::

            [   [master_0 x, master_1 x, master_2 x],
                [master_0 y, master_1 y, master_2 y]
            ]

        If the master values are all the same, we collapse the list to
        as single value instead of a list.

        We then convert this to::

            [ [master_0 x] + [x delta tuple] + [numBlends=1]
              [master_0 y] + [y delta tuple] + [numBlends=1]
            ]
594 def reorder_blend_args(self, commands, get_delta_func):
595 """
596 We first re-order the master coordinate values.
597 For a moveto to lineto, the args are now arranged as::
598
599 [ [master_0 x,y], [master_1 x,y], [master_2 x,y] ]
600
601 We re-arrange this to::
602
603 [ [master_0 x, master_1 x, master_2 x],
604 [master_0 y, master_1 y, master_2 y]
605 ]
606
607 If the master values are all the same, we collapse the list to
608 as single value instead of a list.
609
610 We then convert this to::
611
612 [ [master_0 x] + [x delta tuple] + [numBlends=1]
613 [master_0 y] + [y delta tuple] + [numBlends=1]
614 ]
615 """
616 for cmd in commands:
617 # arg[i] is the set of arguments for this operator from master i.
618 args = cmd[1]
619 m_args = zip(*args)
620 # m_args[n] is now all num_master args for the i'th argument
621 # for this operation.
622 cmd[1] = list(m_args)
623 lastOp = None
624 for cmd in commands:
625 op = cmd[0]
626 # masks are represented by two cmd's: first has only op names,
627 # second has only args.
628 if lastOp in ['hintmask', 'cntrmask']:
629 coord = list(cmd[1])
630 if not allEqual(coord):
631 raise VarLibMergeError("Hintmask values cannot differ between source fonts.")
632 cmd[1] = [coord[0][0]]
633 else:
634 coords = cmd[1]
635 new_coords = []
636 for coord in coords:
637 if allEqual(coord):
638 new_coords.append(coord[0])
639 else:
640 # convert to deltas
641 deltas = get_delta_func(coord)[1:]
642 coord = [coord[0]] + deltas
643 coord.append(1)
644 new_coords.append(coord)
645 cmd[1] = new_coords
646 lastOp = op
647 return commands
