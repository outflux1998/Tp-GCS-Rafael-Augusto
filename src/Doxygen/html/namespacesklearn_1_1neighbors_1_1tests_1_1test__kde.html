17def compute_kernel_slow(Y, X, kernel, h):
18 if h == "scott":
19 h = X.shape[0] ** (-1 / (X.shape[1] + 4))
20 elif h == "silverman":
21 h = (X.shape[0] * (X.shape[1] + 2) / 4) ** (-1 / (X.shape[1] + 4))
22
23 d = np.sqrt(((Y[:, None, :] - X) ** 2).sum(-1))
24 norm = kernel_norm(h, X.shape[1], kernel) / X.shape[0]
25
26 if kernel == "gaussian":
27 return norm * np.exp(-0.5 * (d * d) / (h * h)).sum(-1)
28 elif kernel == "tophat":
29 return norm * (d < h).sum(-1)
30 elif kernel == "epanechnikov":
31 return norm * ((1.0 - (d * d) / (h * h)) * (d < h)).sum(-1)
32 elif kernel == "exponential":
33 return norm * (np.exp(-d / h)).sum(-1)
34 elif kernel == "linear":
35 return norm * ((1 - d / h) * (d < h)).sum(-1)
36 elif kernel == "cosine":
37 return norm * (np.cos(0.5 * np.pi * d / h) * (d < h)).sum(-1)
38 else:
39 raise ValueError("kernel not recognized")
