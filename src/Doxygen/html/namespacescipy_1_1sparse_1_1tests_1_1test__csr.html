116def test_csr_hstack_int64():
117 """
118 Tests if hstack properly promotes to indices and indptr arrays to np.int64
119 when using np.int32 during concatenation would result in either array
120 overflowing.
121 """
122 max_int32 = np.iinfo(np.int32).max
123
124 # First case: indices would overflow with int32
125 data = [1.0]
126 row = [0]
127
128 max_indices_1 = max_int32 - 1
129 max_indices_2 = 3
130
131 # Individual indices arrays are representable with int32
132 col_1 = [max_indices_1 - 1]
133 col_2 = [max_indices_2 - 1]
134
135 X_1 = csr_matrix((data, (row, col_1)))
136 X_2 = csr_matrix((data, (row, col_2)))
137
138 assert max(max_indices_1 - 1, max_indices_2 - 1) < max_int32
139 assert X_1.indices.dtype == X_1.indptr.dtype == np.int32
140 assert X_2.indices.dtype == X_2.indptr.dtype == np.int32
141
142 # ... but when concatenating their CSR matrices, the resulting indices
143 # array can't be represented with int32 and must be promoted to int64.
144 X_hs = hstack([X_1, X_2], format="csr")
145
146 assert X_hs.indices.max() == max_indices_1 + max_indices_2 - 1
147 assert max_indices_1 + max_indices_2 - 1 > max_int32
148 assert X_hs.indices.dtype == X_hs.indptr.dtype == np.int64
149
150 # Even if the matrices are empty, we must account for their size
151 # contribution so that we may safely set the final elements.
152 X_1_empty = csr_matrix(X_1.shape)
153 X_2_empty = csr_matrix(X_2.shape)
154 X_hs_empty = hstack([X_1_empty, X_2_empty], format="csr")
155
156 assert X_hs_empty.shape == X_hs.shape
157 assert X_hs_empty.indices.dtype == np.int64
158
159 # Should be just small enough to stay in int32 after stack. Note that
160 # we theoretically could support indices.max() == max_int32, but due to an
161 # edge-case in the underlying sparsetools code
162 # (namely the `coo_tocsr` routine),
163 # we require that max(X_hs_32.shape) < max_int32 as well.
164 # Hence we can only support max_int32 - 1.
165 col_3 = [max_int32 - max_indices_1 - 1]
166 X_3 = csr_matrix((data, (row, col_3)))
167 X_hs_32 = hstack([X_1, X_3], format="csr")
168 assert X_hs_32.indices.dtype == np.int32
169 assert X_hs_32.indices.max() == max_int32 - 1
