155def test_quiver_key_xy():
156 # With scale_units='xy', ensure quiverkey still matches its quiver.
157 # Note that the quiver and quiverkey lengths depend on the axes aspect
158 # ratio, and that with angles='xy' their angles also depend on the axes
159 # aspect ratio.
160 X = np.arange(8)
161 Y = np.zeros(8)
162 angles = X * (np.pi / 4)
163 uv = np.exp(1j * angles)
164 U = uv.real
165 V = uv.imag
166 fig, axs = plt.subplots(2)
167 for ax, angle_str in zip(axs, ('uv', 'xy')):
168 ax.set_xlim(-1, 8)
169 ax.set_ylim(-0.2, 0.2)
170 q = ax.quiver(X, Y, U, V, pivot='middle',
171 units='xy', width=0.05,
172 scale=2, scale_units='xy',
173 angles=angle_str)
174 for x, angle in zip((0.2, 0.5, 0.8), (0, 45, 90)):
175 ax.quiverkey(q, X=x, Y=0.8, U=1, angle=angle, label='', color='b')
