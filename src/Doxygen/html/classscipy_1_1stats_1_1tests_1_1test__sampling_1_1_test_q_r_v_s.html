378 def test_QRVS_size_tuple(self, method):
379 # QMCEngine samples are always of shape (n, d). When `size` is a tuple,
380 # we set `n = prod(size)` in the call to qmc_engine.random, transform
381 # the sample, and reshape it to the final dimensions. When we reshape,
382 # we need to be careful, because the _columns_ of the sample returned
383 # by a QMCEngine are "independent"-ish, but the elements within the
384 # columns are not. We need to make sure that this doesn't get mixed up
385 # by reshaping: qrvs[..., i] should remain "independent"-ish of
386 # qrvs[..., i+1], but the elements within qrvs[..., i] should be
387 # transformed from the same low-discrepancy sequence.
388
389 dist = StandardNormal()
390 Method = getattr(stats.sampling, method)
391 gen = Method(dist)

394 d = 5
395 qrng = stats.qmc.Halton(d, seed=0)
396 qrng2 = stats.qmc.Halton(d, seed=0)
397
398 uniform = qrng2.random(np.prod(size))
399
