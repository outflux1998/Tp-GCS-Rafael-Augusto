79 # This test is heavily inspired from test_random.py of python-core.
80 #
81 # For the entire allowable range of 0 <= k <= N, validate that
82 # sample generates all possible permutations
83 n_population = 10
84
85 # a large number of trials prevents false negatives without slowing normal
86 # case
87 n_trials = 10000
88
89 for n_samples in range(n_population):
90 # Counting the number of combinations is not as good as counting the
91 # the number of permutations. However, it works with sampling algorithm
92 # that does not provide a random permutation of the subset of integer.
93 n_expected = comb(n_population, n_samples, exact=True)
94
95 output = {}

99 ] = None

103 else:
104 raise AssertionError(
