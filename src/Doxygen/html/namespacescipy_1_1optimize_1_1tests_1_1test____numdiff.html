721 # test for gh12487
722 # if an absolute step is specified for 2-point differences make sure that
723 # the side corresponds to the step. i.e. if step is positive then forward
724 # differences should be used, if step is negative then backwards
725 # differences should be used.
726
727 # function has double discontinuity at x = [-1, -1]
728 # first component is \/, second component is /\
729 def f(x):
730 return -np.abs(x[0] + 1) + np.abs(x[1] + 1)
731
732 # check that the forward difference is used
733 grad = approx_derivative(f, [-1, -1], method='2-point', abs_step=1e-8)
734 assert_allclose(grad, [-1.0, 1.0])
735
736 # check that the backwards difference is used
737 grad = approx_derivative(f, [-1, -1], method='2-point', abs_step=-1e-8)
738 assert_allclose(grad, [1.0, -1.0])
739
740 # check that the forwards difference is used with a step for both
741 # parameters
742 grad = approx_derivative(
743 f, [-1, -1], method='2-point', abs_step=[1e-8, 1e-8]
744 )
745 assert_allclose(grad, [-1.0, 1.0])
746
747 # check that we can mix forward/backwards steps.
748 grad = approx_derivative(
749 f, [-1, -1], method='2-point', abs_step=[1e-8, -1e-8]
750 )
751 assert_allclose(grad, [-1.0, -1.0])
752 grad = approx_derivative(
753 f, [-1, -1], method='2-point', abs_step=[-1e-8, 1e-8]
754 )
755 assert_allclose(grad, [1.0, 1.0])
