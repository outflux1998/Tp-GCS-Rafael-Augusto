132 # Test matrices from `illc1850.coord` and `mhd1280b.cua` distributed with
133 # PROPACK 2.1: http://sun.stanford.edu/~rmunk/PROPACK/

147 # complex example matrix has many repeated singular values, so check only
148 # beginning non-repeated singular vectors to avoid permutations
149 sv_check = 27 if precision in {'complex8', 'complex16'} else k
150 u = u[:, :sv_check]
151 vh = vh[:sv_check, :]

154 # Check orthogonality of singular vectors
155 assert_allclose(np.eye(u.shape[1]), u.conj().T @ u, atol=atol)

159 # PROPACK reconstructed matrices is small
