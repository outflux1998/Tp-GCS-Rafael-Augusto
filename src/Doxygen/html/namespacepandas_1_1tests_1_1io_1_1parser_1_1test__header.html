543def test_multi_index_unnamed(all_parsers, index_col, columns):
544 # see gh-23687
545 #
546 # When specifying a multi-index header, make sure that
547 # we don't error just because one of the rows in our header
548 # has ALL column names containing the string "Unnamed". The
549 # correct condition to check is whether the row contains
550 # ALL columns that did not have names (and instead were given
551 # placeholder ones).
552 parser = all_parsers
553 header = [0, 1]
554
555 if index_col is None:
556 data = ",".join(columns or ["", ""]) + "\n0,1\n2,3\n4,5\n"
557 else:
558 data = ",".join([""] + (columns or ["", ""])) + "\n,0,1\n0,2,3\n1,4,5\n"
559
560 result = parser.read_csv(StringIO(data), header=header, index_col=index_col)
561 exp_columns = []
562
563 if columns is None:
564 columns = ["", "", ""]
565
566 for i, col in enumerate(columns):
567 if not col: # Unnamed.
568 col = f"Unnamed: {i if index_col is None else i + 1}_level_0"
569
570 exp_columns.append(col)
571
572 columns = MultiIndex.from_tuples(zip(exp_columns, ["0", "1"]))
573 expected = DataFrame([[2, 3], [4, 5]], columns=columns)
574 tm.assert_frame_equal(result, expected)
