247 def test_int64_uint64_corner_case(self):
248 # When stored in Numpy arrays, `lbnd` is casted
249 # as np.int64, and `ubnd` is casted as np.uint64.
250 # Checking whether `lbnd` >= `ubnd` used to be
251 # done solely via direct comparison, which is incorrect
252 # because when Numpy tries to compare both numbers,
253 # it casts both to np.float64 because there is
254 # no integer superset of np.int64 and np.uint64. However,
255 # `ubnd` is too large to be represented in np.float64,
256 # causing it be round down to np.iinfo(np.int64).max,
257 # leading to a ValueError because `lbnd` now equals
258 # the new `ubnd`.
259
260 dt = np.int64
261 tgt = np.iinfo(np.int64).max
262 lbnd = np.int64(np.iinfo(np.int64).max)
263 ubnd = np.uint64(np.iinfo(np.int64).max + 1)
264
265 # None of these function calls should
266 # generate a ValueError now.
267 actual = np.random.randint(lbnd, ubnd, dtype=dt)
268 assert_equal(actual, tgt)
