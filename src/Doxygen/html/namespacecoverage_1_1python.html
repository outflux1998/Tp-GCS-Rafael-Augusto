100def source_for_file(filename: str) -> str:
101 """Return the source filename for `filename`.
102
103 Given a file name being traced, return the best guess as to the source
104 file to attribute it to.
105
106 """
107 if filename.endswith(".py"):
108 # .py files are themselves source files.
109 return filename
110
111 elif filename.endswith((".pyc", ".pyo")):
112 # Bytecode files probably have source files near them.
113 py_filename = filename[:-1]
114 if os.path.exists(py_filename):
