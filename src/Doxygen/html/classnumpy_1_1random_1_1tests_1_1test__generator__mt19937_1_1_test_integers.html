611 def test_int64_uint64_corner_case(self, endpoint):
612 # When stored in Numpy arrays, `lbnd` is casted
613 # as np.int64, and `ubnd` is casted as np.uint64.
614 # Checking whether `lbnd` >= `ubnd` used to be
615 # done solely via direct comparison, which is incorrect
616 # because when Numpy tries to compare both numbers,
617 # it casts both to np.float64 because there is
618 # no integer superset of np.int64 and np.uint64. However,
619 # `ubnd` is too large to be represented in np.float64,
620 # causing it be round down to np.iinfo(np.int64).max,
621 # leading to a ValueError because `lbnd` now equals
622 # the new `ubnd`.
623
624 dt = np.int64
625 tgt = np.iinfo(np.int64).max
626 lbnd = np.int64(np.iinfo(np.int64).max)
627 ubnd = np.uint64(np.iinfo(np.int64).max + 1 - endpoint)
628
629 # None of these function calls should
630 # generate a ValueError now.
631 actual = random.integers(lbnd, ubnd, endpoint=endpoint, dtype=dt)
632 assert_equal(actual, tgt)
