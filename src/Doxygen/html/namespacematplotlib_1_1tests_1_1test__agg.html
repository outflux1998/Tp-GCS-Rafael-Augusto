89 def smooth1d(x, window_len):
90 # copied from https://scipy-cookbook.readthedocs.io/
91 s = np.r_[
92 2*x[0] - x[window_len:1:-1], x, 2*x[-1] - x[-1:-window_len:-1]]
93 w = np.hanning(window_len)
94 y = np.convolve(w/w.sum(), s, mode='same')
95 return y[window_len-1:-window_len+1]
96
97 def smooth2d(A, sigma=3):
98 window_len = max(int(sigma), 3) * 2 + 1
99 A = np.apply_along_axis(smooth1d, 0, A, window_len)
100 A = np.apply_along_axis(smooth1d, 1, A, window_len)
101 return A

279 # NOTE: This test covers multiple independent test scenarios in a single
280 # function, because each scenario uses ~2GB of memory and we don't
281 # want parallel test executors to accidentally run multiple of these
282 # at the same time.
283
284 N = 100_000
285 dpi = 500
286 w = 5*dpi
287 h = 6*dpi
288
289 # make a Path that spans the whole w-h rectangle
290 x = np.linspace(0, w, N)
291 y = np.ones(N) * h
292 y[::2] = 0
293 path = Path(np.vstack((x, y)).T)
294 # effectively disable path simplification (but leaving it "on")
295 path.simplify_threshold = 0
296
297 # setup the minimal GraphicsContext to draw a Path
298 ra = RendererAgg(w, h, dpi)
299 gc = ra.new_gc()
300 gc.set_linewidth(1)
301 gc.set_foreground('r')
302
303 gc.set_hatch('/')
304 with pytest.raises(OverflowError, match='can not split hatched path'):
305 ra.draw_path(gc, path, IdentityTransform())
306 gc.set_hatch(None)
307
308 with pytest.raises(OverflowError, match='can not split filled path'):
309 ra.draw_path(gc, path, IdentityTransform(), (1, 0, 0))
310
311 # Set to zero to disable, currently defaults to 0, but let's be sure.
312 with rc_context({'agg.path.chunksize': 0}):
313 with pytest.raises(OverflowError, match='Please set'):
314 ra.draw_path(gc, path, IdentityTransform())
315
316 # Set big enough that we do not try to chunk.
317 with rc_context({'agg.path.chunksize': 1_000_000}):
318 with pytest.raises(OverflowError, match='Please reduce'):
319 ra.draw_path(gc, path, IdentityTransform())
320
321 # Small enough we will try to chunk, but big enough we will fail to render.
322 with rc_context({'agg.path.chunksize': 90_000}):
323 with pytest.raises(OverflowError, match='Please reduce'):
324 ra.draw_path(gc, path, IdentityTransform())
