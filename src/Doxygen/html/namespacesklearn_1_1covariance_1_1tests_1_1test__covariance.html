198def _naive_ledoit_wolf_shrinkage(X):
199 # A simple implementation of the formulas from Ledoit & Wolf
200
201 # The computation below achieves the following computations of the
202 # "O. Ledoit and M. Wolf, A Well-Conditioned Estimator for
203 # Large-Dimensional Covariance Matrices"
204 # beta and delta are given in the beginning of section 3.2
205 n_samples, n_features = X.shape
206 emp_cov = empirical_covariance(X, assume_centered=False)
207 mu = np.trace(emp_cov) / n_features
208 delta_ = emp_cov.copy()
209 delta_.flat[:: n_features + 1] -= mu
210 delta = (delta_**2).sum() / n_features
211 X2 = X**2
212 beta_ = (
213 1.0
214 / (n_features * n_samples)
215 * np.sum(np.dot(X2.T, X2) / n_samples - emp_cov**2)
216 )
217
218 beta = min(beta_, delta)
219 shrinkage = beta / delta
220 return shrinkage
