136 # Shift and scale the polynomial domain to be between -1 and 1
137 mins = np.min(y, axis=0)
138 maxs = np.max(y, axis=0)
139 shift = (maxs + mins)/2
140 scale = (maxs - mins)/2
141 # The scale may be zero if there is a single point or all the points have
