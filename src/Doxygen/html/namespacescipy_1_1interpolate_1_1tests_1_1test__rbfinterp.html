42def _is_conditionally_positive_definite(kernel, m):
43 # Tests whether the kernel is conditionally positive definite of order m.
44 # See chapter 7 of Fasshauer's "Meshfree Approximation Methods with
45 # MATLAB".
46 nx = 10
47 ntests = 100
48 for ndim in [1, 2, 3, 4, 5]:
49 # Generate sample points with a Halton sequence to avoid samples that
50 # are too close to eachother, which can make the matrix singular.
51 seq = Halton(ndim, scramble=False, seed=np.random.RandomState())
52 for _ in range(ntests):
53 x = 2*seq.random(nx) - 1
54 A = _rbfinterp_pythran._kernel_matrix(x, kernel)
55 P = _vandermonde(x, m - 1)
56 Q, R = np.linalg.qr(P, mode='complete')
57 # Q2 forms a basis spanning the space where P.T.dot(x) = 0. Project
58 # A onto this space, and then see if it is positive definite using
59 # the Cholesky decomposition. If not, then the kernel is not c.p.d.
60 # of order m.
61 Q2 = Q[:, P.shape[1]:]
62 B = Q2.T.dot(A).dot(Q2)
63 try:
64 np.linalg.cholesky(B)
65 except np.linalg.LinAlgError:
66 return False
67
68 return True
