74 def ip(x, y):
75 t1 = (x + y <= 1)
76 t2 = ~t1
77
78 x1 = x[t1]
79 y1 = y[t1]
80
81 x2 = x[t2]
82 y2 = y[t2]
83
84 z = 0*x
85
86 z[t1] = (values[0]*(1 - x1 - y1)
87 + values[1]*y1
88 + values[3]*x1)
89
90 z[t2] = (values[2]*(x2 + y2 - 1)
91 + values[1]*(1 - x2)
92 + values[3]*(1 - y2))
93 return z
