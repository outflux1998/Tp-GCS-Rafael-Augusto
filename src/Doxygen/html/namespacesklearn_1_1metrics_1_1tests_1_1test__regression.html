541def test_mean_pinball_loss_on_constant_predictions(distribution, target_quantile):
542 if not hasattr(np, "quantile"):
543 pytest.skip(
544 "This test requires a more recent version of numpy "
545 "with support for np.quantile."
546 )
547
548 # Check that the pinball loss is minimized by the empirical quantile.
549 n_samples = 3000
550 rng = np.random.RandomState(42)
551 data = getattr(rng, distribution)(size=n_samples)
552
553 # Compute the best possible pinball loss for any constant predictor:
554 best_pred = np.quantile(data, target_quantile)
555 best_constant_pred = np.full(n_samples, fill_value=best_pred)
556 best_pbl = mean_pinball_loss(data, best_constant_pred, alpha=target_quantile)
557
558 # Evaluate the loss on a grid of quantiles
559 candidate_predictions = np.quantile(data, np.linspace(0, 1, 100))
560 for pred in candidate_predictions:
561 # Compute the pinball loss of a constant predictor:
562 constant_pred = np.full(n_samples, fill_value=pred)
563 pbl = mean_pinball_loss(data, constant_pred, alpha=target_quantile)
564
565 # Check that the loss of this constant predictor is greater or equal
566 # than the loss of using the optimal quantile (up to machine
567 # precision):
568 assert pbl >= best_pbl - np.finfo(best_pbl.dtype).eps
569
570 # Check that the value of the pinball loss matches the analytical
571 # formula.
572 expected_pbl = (pred - data[data < pred]).sum() * (1 - target_quantile) + (
573 data[data >= pred] - pred
574 ).sum() * target_quantile
575 expected_pbl /= n_samples
576 assert_almost_equal(expected_pbl, pbl)
577
578 # Check that we can actually recover the target_quantile by minimizing the
579 # pinball loss w.r.t. the constant prediction quantile.
580 def objective_func(x):
581 constant_pred = np.full(n_samples, fill_value=x)
582 return mean_pinball_loss(data, constant_pred, alpha=target_quantile)
583
584 result = optimize.minimize(objective_func, data.mean(), method="Nelder-Mead")
585 assert result.success
586 # The minimum is not unique with limited data, hence the large tolerance.
587 assert result.x == pytest.approx(best_pred, rel=1e-2)
588 assert result.fun == pytest.approx(best_pbl)
