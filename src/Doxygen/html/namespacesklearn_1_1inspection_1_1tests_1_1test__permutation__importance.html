393def test_permutation_importance_sample_weight():
394 # Creating data with 2 features and 1000 samples, where the target
395 # variable is a linear combination of the two features, such that
396 # in half of the samples the impact of feature 1 is twice the impact of
397 # feature 2, and vice versa on the other half of the samples.
398 rng = np.random.RandomState(1)
399 n_samples = 1000
400 n_features = 2
401 n_half_samples = n_samples // 2
402 x = rng.normal(0.0, 0.001, (n_samples, n_features))
403 y = np.zeros(n_samples)
404 y[:n_half_samples] = 2 * x[:n_half_samples, 0] + x[:n_half_samples, 1]
405 y[n_half_samples:] = x[n_half_samples:, 0] + 2 * x[n_half_samples:, 1]
406
407 # Fitting linear regression with perfect prediction
408 lr = LinearRegression(fit_intercept=False)
409 lr.fit(x, y)
410
411 # When all samples are weighted with the same weights, the ratio of
412 # the two features importance should equal to 1 on expectation (when using
413 # mean absolutes error as the loss function).
414 pi = permutation_importance(
415 lr, x, y, random_state=1, scoring="neg_mean_absolute_error", n_repeats=200
416 )
417 x1_x2_imp_ratio_w_none = pi.importances_mean[0] / pi.importances_mean[1]
418 assert x1_x2_imp_ratio_w_none == pytest.approx(1, 0.01)
419
420 # When passing a vector of ones as the sample_weight, results should be
421 # the same as in the case that sample_weight=None.
422 w = np.ones(n_samples)
423 pi = permutation_importance(
424 lr,
425 x,
426 y,
427 random_state=1,
428 scoring="neg_mean_absolute_error",
429 n_repeats=200,
430 sample_weight=w,
431 )
432 x1_x2_imp_ratio_w_ones = pi.importances_mean[0] / pi.importances_mean[1]
433 assert x1_x2_imp_ratio_w_ones == pytest.approx(x1_x2_imp_ratio_w_none, 0.01)
434
435 # When the ratio between the weights of the first half of the samples and
436 # the second half of the samples approaches to infinity, the ratio of
437 # the two features importance should equal to 2 on expectation (when using
438 # mean absolutes error as the loss function).
439 w = np.hstack(
440 [np.repeat(10.0**10, n_half_samples), np.repeat(1.0, n_half_samples)]
441 )
442 lr.fit(x, y, w)
443 pi = permutation_importance(
444 lr,
445 x,
446 y,
447 random_state=1,
448 scoring="neg_mean_absolute_error",
449 n_repeats=200,
450 sample_weight=w,
451 )

108def test_robustness_to_high_cardinality_noisy_feature(n_jobs, max_samples, seed=42):
109 # Permutation variable importance should not be affected by the high
110 # cardinality bias of traditional feature importances, especially when
111 # computed on a held-out test set:
112 rng = np.random.RandomState(seed)
113 n_repeats = 5
114 n_samples = 1000
115 n_classes = 5
116 n_informative_features = 2
117 n_noise_features = 1
118 n_features = n_informative_features + n_noise_features
119
120 # Generate a multiclass classification dataset and a set of informative
121 # binary features that can be used to predict some classes of y exactly
122 # while leaving some classes unexplained to make the problem harder.
123 classes = np.arange(n_classes)
124 y = rng.choice(classes, size=n_samples)
125 X = np.hstack([(y == c).reshape(-1, 1) for c in classes[:n_informative_features]])
126 X = X.astype(np.float32)
127
128 # Not all target classes are explained by the binary class indicator
129 # features:
130 assert n_informative_features < n_classes
131
132 # Add 10 other noisy features with high cardinality (numerical) values
133 # that can be used to overfit the training data.
134 X = np.concatenate([X, rng.randn(n_samples, n_noise_features)], axis=1)
135 assert X.shape == (n_samples, n_features)
136
137 # Split the dataset to be able to evaluate on a held-out test set. The
138 # Test size should be large enough for importance measurements to be
139 # stable:
140 X_train, X_test, y_train, y_test = train_test_split(
141 X, y, test_size=0.5, random_state=rng
142 )
143 clf = RandomForestClassifier(n_estimators=5, random_state=rng)
144 clf.fit(X_train, y_train)
145
146 # Variable importances computed by impurity decrease on the tree node
147 # splits often use the noisy features in splits. This can give misleading
148 # impression that high cardinality noisy variables are the most important:
149 tree_importances = clf.feature_importances_
150 informative_tree_importances = tree_importances[:n_informative_features]
151 noisy_tree_importances = tree_importances[n_informative_features:]
152 assert informative_tree_importances.max() < noisy_tree_importances.min()
153
154 # Let's check that permutation-based feature importances do not have this
155 # problem.
156 r = permutation_importance(
157 clf,
158 X_test,
159 y_test,
160 n_repeats=n_repeats,
161 random_state=rng,
162 n_jobs=n_jobs,
163 max_samples=max_samples,
164 )
165
166 assert r.importances.shape == (X.shape[1], n_repeats)
167
168 # Split the importances between informative and noisy features
169 informative_importances = r.importances_mean[:n_informative_features]
170 noisy_importances = r.importances_mean[n_informative_features:]
171
172 # Because we do not have a binary variable explaining each target classes,
173 # the RF model will have to use the random variable to make some
174 # (overfitting) splits (as max_depth is not set). Therefore the noisy
175 # variables will be non-zero but with small values oscillating around
176 # zero:
177 assert max(np.abs(noisy_importances)) > 1e-7
178 assert noisy_importances.max() < 0.05
179
180 # The binary features correlated with y should have a higher importance
181 # than the high cardinality noisy features.
182 # The maximum test accuracy is 2 / 5 == 0.4, each informative feature
183 # contributing approximately a bit more than 0.2 of accuracy.
184 assert informative_importances.min() > 0.15
