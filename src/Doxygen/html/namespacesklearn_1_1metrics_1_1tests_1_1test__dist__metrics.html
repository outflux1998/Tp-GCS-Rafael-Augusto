315 # Haversine is not supported by scipy.special.distance.{cdist,pdist}
316 # So we reimplement it to have a reference.
317 def haversine_slow(x1, x2):
318 return 2 * np.arcsin(
319 np.sqrt(
320 np.sin(0.5 * (x1[0] - x2[0])) ** 2
321 + np.cos(x1[0]) * np.cos(x2[0]) * np.sin(0.5 * (x1[1] - x2[1])) ** 2
322 )
323 )
324
325 D_reference = np.zeros((X_csr.shape[0], Y_csr.shape[0]))
326 for i, xi in enumerate(X):
327 for j, yj in enumerate(Y):
328 D_reference[i, j] = haversine_slow(xi, yj)
329

331
332 D_sklearn = haversine.pairwise(X, Y)

334 haversine.dist_to_rdist(D_sklearn), np.sin(0.5 * D_reference) ** 2, rtol=1e-6
