250 # GH14856
251 # Test various combinations of string slicing resolution vs.
252 # index resolution
253 # - If string resolution is less precise than index resolution,
254 # string is considered a slice
255 # - If string resolution is equal to or more precise than index
256 # resolution, string is considered an exact match
257 formats = [
258 "%Y",
259 "%Y-%m",
260 "%Y-%m-%d",
261 "%Y-%m-%d %H",
262 "%Y-%m-%d %H:%M",
263 "%Y-%m-%d %H:%M:%S",
264 ]
265 resolutions = ["year", "month", "day", "hour", "minute", "second"]
266 for rnum, resolution in enumerate(resolutions[2:], 2):
267 # we check only 'day', 'hour', 'minute' and 'second'
268 unit = Timedelta("1 " + resolution)
269 middate = datetime(2012, 1, 1, 0, 0, 0)
270 index = DatetimeIndex([middate - unit, middate, middate + unit])
271 values = [1, 2, 3]
272 df = DataFrame({"a": values}, index, dtype=np.int64)
273 assert df.index.resolution == resolution
274
275 # Timestamp with the same resolution as index
276 # Should be exact match for Series (return scalar)
277 # and raise KeyError for Frame
278 for timestamp, expected in zip(index, values):
279 ts_string = timestamp.strftime(formats[rnum])
280 # make ts_string as precise as index
281 result = df["a"][ts_string]
282 assert isinstance(result, np.int64)
283 assert result == expected
284 msg = rf"^'{ts_string}'$"
285 with pytest.raises(KeyError, match=msg):
286 df[ts_string]
287
288 # Timestamp with resolution less precise than index
289 for fmt in formats[:rnum]:
290 for element, theslice in [[0, slice(None, 1)], [1, slice(1, None)]]:
291 ts_string = index[element].strftime(fmt)
292
293 # Series should return slice
294 result = df["a"][ts_string]
295 expected = df["a"][theslice]
296 tm.assert_series_equal(result, expected)
297
298 # Frame should return slice as well
299 with tm.assert_produces_warning(FutureWarning):
300 # GH#36179 deprecated this indexing
301 result = df[ts_string]
302 expected = df[theslice]
303 tm.assert_frame_equal(result, expected)
304
305 # Timestamp with resolution more precise than index
306 # Compatible with existing key
307 # Should return scalar for Series
308 # and raise KeyError for Frame
309 for fmt in formats[rnum + 1 :]:
310 ts_string = index[1].strftime(fmt)
311 result = df["a"][ts_string]
312 assert isinstance(result, np.int64)
313 assert result == 2
314 msg = rf"^'{ts_string}'$"
315 with pytest.raises(KeyError, match=msg):
316 df[ts_string]
317
318 # Not compatible with existing key
319 # Should raise KeyError
320 for fmt, res in list(zip(formats, resolutions))[rnum + 1 :]:
321 ts = index[1] + Timedelta("1 " + res)
322 ts_string = ts.strftime(fmt)
323 msg = rf"^'{ts_string}'$"
324 with pytest.raises(KeyError, match=msg):
325 df["a"][ts_string]
326 with pytest.raises(KeyError, match=msg):
