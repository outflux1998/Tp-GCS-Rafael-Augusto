136def test_fht_exact(n):
137 rng = np.random.RandomState(3491349965)
138
139 # for a(r) a power law r^\gamma, the fast Hankel transform produces the
140 # exact continuous Hankel transform if biased with q = \gamma
141
142 mu = rng.uniform(0, 3)
143
144 # convergence of HT: -1-mu < gamma < 1/2
145 gamma = rng.uniform(-1-mu, 1/2)
146
147 r = np.logspace(-2, 2, n)
148 a = r**gamma
149
150 dln = np.log(r[1]/r[0])
151
152 offset = fhtoffset(dln, mu, initial=0.0, bias=gamma)
153
154 A = fht(a, dln, mu, offset=offset, bias=gamma)
155
156 k = np.exp(offset)/r[::-1]
157
158 # analytical result
159 At = (2/k)**gamma * poch((mu+1-gamma)/2, gamma)
160
161 assert_allclose(A, At)
