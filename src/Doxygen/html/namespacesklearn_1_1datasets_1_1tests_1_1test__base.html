315def test_bunch_pickle_generated_with_0_16_and_read_with_0_17():
316 bunch = Bunch(key="original")
317 # This reproduces a problem when Bunch pickles have been created
318 # with scikit-learn 0.16 and are read with 0.17. Basically there
319 # is a surprising behaviour because reading bunch.key uses
320 # bunch.__dict__ (which is non empty for 0.16 Bunch objects)
321 # whereas assigning into bunch.key uses bunch.__setattr__. See
322 # https://github.com/scikit-learn/scikit-learn/issues/6196 for
323 # more details
324 bunch.__dict__["key"] = "set from __dict__"
325 bunch_from_pkl = loads(dumps(bunch))
326 # After loading from pickle the __dict__ should have been ignored
327 assert bunch_from_pkl.key == "original"
328 assert bunch_from_pkl["key"] == "original"
329 # Making sure that changing the attr does change the value
330 # associated with __getitem__ as well
331 bunch_from_pkl.key = "changed"
