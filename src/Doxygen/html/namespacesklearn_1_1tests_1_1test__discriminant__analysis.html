327def test_lda_orthogonality():
328 # arrange four classes with their means in a kite-shaped pattern
329 # the longer distance should be transformed to the first component, and
330 # the shorter distance to the second component.
331 means = np.array([[0, 0, -1], [0, 2, 0], [0, -2, 0], [0, 0, 5]])
332
333 # We construct perfectly symmetric distributions, so the LDA can estimate
334 # precise means.
335 scatter = np.array(
336 [
337 [0.1, 0, 0],
338 [-0.1, 0, 0],
339 [0, 0.1, 0],
340 [0, -0.1, 0],
341 [0, 0, 0.1],
342 [0, 0, -0.1],
343 ]
344 )
345
346 X = (means[:, np.newaxis, :] + scatter[np.newaxis, :, :]).reshape((-1, 3))
347 y = np.repeat(np.arange(means.shape[0]), scatter.shape[0])
348
349 # Fit LDA and transform the means
350 clf = LinearDiscriminantAnalysis(solver="svd").fit(X, y)
351 means_transformed = clf.transform(means)
352
353 d1 = means_transformed[3] - means_transformed[0]
354 d2 = means_transformed[2] - means_transformed[1]
355 d1 /= np.sqrt(np.sum(d1**2))
356 d2 /= np.sqrt(np.sum(d2**2))
357
358 # the transformed within-class covariance should be the identity matrix
359 assert_almost_equal(np.cov(clf.transform(scatter).T), np.eye(2))
360
361 # the means of classes 0 and 3 should lie on the first component
362 assert_almost_equal(np.abs(np.dot(d1[:2], [1, 0])), 1.0)
363
364 # the means of classes 1 and 2 should lie on the second component
365 assert_almost_equal(np.abs(np.dot(d2[:2], [0, 1])), 1.0)

141 def generate_dataset(n_samples, centers, covariances, random_state=None):
142 """Generate a multivariate normal data given some centers and
143 covariances"""
144 rng = check_random_state(random_state)
145 X = np.vstack(
146 [
147 rng.multivariate_normal(mean, cov, size=n_samples // len(centers))
148 for mean, cov in zip(centers, covariances)
149 ]
150 )
151 y = np.hstack(
152 [[clazz] * (n_samples // len(centers)) for clazz in range(len(centers))]
153 )
154 return X, y
155
156 blob_centers = np.array([[0, 0], [-10, 40], [-30, 30]])[:n_classes]
157 blob_stds = np.array([[[10, 10], [10, 100]]] * len(blob_centers))
158 X, y = generate_dataset(
159 n_samples=90000, centers=blob_centers, covariances=blob_stds, random_state=42
160 )
161 lda = LinearDiscriminantAnalysis(
162 solver=solver, store_covariance=True, shrinkage=None
163 ).fit(X, y)
164 # check that the empirical means and covariances are close enough to the
165 # one used to generate the data
166 assert_allclose(lda.means_, blob_centers, atol=1e-1)
167 assert_allclose(lda.covariance_, blob_stds[0], atol=1)
168
169 # implement the method to compute the probability given in The Elements
170 # of Statistical Learning (cf. p.127, Sect. 4.4.5 "Logistic Regression
171 # or LDA?")
172 precision = linalg.inv(blob_stds[0])
173 alpha_k = []
174 alpha_k_0 = []
175 for clazz in range(len(blob_centers) - 1):
176 alpha_k.append(
177 np.dot(precision, (blob_centers[clazz] - blob_centers[-1])[:, np.newaxis])
178 )
179 alpha_k_0.append(
180 np.dot(
181 -0.5 * (blob_centers[clazz] + blob_centers[-1])[np.newaxis, :],
182 alpha_k[-1],
183 )
184 )
185
186 sample = np.array([[-22, 22]])
187
188 def discriminant_func(sample, coef, intercept, clazz):
189 return np.exp(intercept[clazz] + np.dot(sample, coef[clazz]))
190
191 prob = np.array(
192 [
193 float(
194 discriminant_func(sample, alpha_k, alpha_k_0, clazz)
195 / (
196 1
197 + sum(
198 [
199 discriminant_func(sample, alpha_k, alpha_k_0, clazz)
200 for clazz in range(n_classes - 1)
201 ]
202 )
203 )
204 )
205 for clazz in range(n_classes - 1)
206 ]
207 )
208
209 prob_ref = 1 - np.sum(prob)
210
211 # check the consistency of the computed probability
212 # all probabilities should sum to one
213 prob_ref_2 = float(
214 1
215 / (
216 1
217 + sum(
218 [
219 discriminant_func(sample, alpha_k, alpha_k_0, clazz)
220 for clazz in range(n_classes - 1)
221 ]
222 )
223 )
224 )
225
226 assert prob_ref == pytest.approx(prob_ref_2)
227 # check that the probability of LDA are close to the theoretical
228 # probabilties
229 assert_allclose(
230 lda.predict_proba(sample), np.hstack([prob, prob_ref])[np.newaxis], atol=1e-2
231 )
