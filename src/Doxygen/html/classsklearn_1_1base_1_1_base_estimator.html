Detailed Description

Base class for all estimators in scikit-learn.

Notes
-----
All estimators should specify all the parameters that can be set
at the class level in their ``__init__`` as explicit keyword
arguments (no ``*args`` or ``**kwargs``).

125 # fetch the constructor or the original constructor before
126 # deprecation wrapping if any
127 init = getattr(cls.__init__, "deprecated_original", cls.__init__)
128 if init is object.__init__:
129 # No explicit constructor to introspect
130 return []
131
132 # introspect the constructor arguments to find the model parameters
133 # to represent
134 init_signature = inspect.signature(init)
135 # Consider the constructor parameters excluding 'self'
136 parameters = [
137 p
138 for p in init_signature.parameters.values()
139 if p.name != "self" and p.kind != p.VAR_KEYWORD
140 ]
141 for p in parameters:
142 if p.kind == p.VAR_POSITIONAL:
143 raise RuntimeError(
144 "scikit-learn estimators should always "
145 "specify their parameters in the signature"
146 " of their __init__ (no varargs)."
147 " %s with constructor %s doesn't "
148 " follow this convention." % (cls, init_signature)
149 )
150 # Extract and sort argument names excluding 'self'
151 return sorted([p.name for p in parameters])
