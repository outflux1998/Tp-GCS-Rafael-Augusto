462def test_maxiter_worsening(solver):
463 # Check error does not grow (boundlessly) with increasing maxiter.
464 # This can occur due to the solvers hitting close to breakdown,
465 # which they should detect and halt as necessary.
466 # cf. gh-9100
467
468 # Singular matrix, rhs numerically not in range

470 [0, -0.13627952880333782+6.283185307179586j, 0, 0],
471 [0, 0, -0.13627952880333782-6.283185307179586j, 0],
472 [0.1112795288033368, 0j, 0j, -0.16127952880333785]])
473 v = np.ones(4)
474 best_error = np.inf
475 tol = 7 if platform.machine() == 'aarch64' else 5
476
477 for maxiter in range(1, 20):
478 x, info = solver(A, v, maxiter=maxiter, tol=1e-8, atol=0)
479

481 assert_(np.linalg.norm(A.dot(x) - v) <= 1e-8*np.linalg.norm(v))
482

484 best_error = min(best_error, error)
