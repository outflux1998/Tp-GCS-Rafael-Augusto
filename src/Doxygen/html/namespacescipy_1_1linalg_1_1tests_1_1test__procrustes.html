77def test_orthogonal_procrustes():
78 np.random.seed(1234)
79 for m, n in ((6, 4), (4, 4), (4, 6)):
80 # Sample a random target matrix.
81 B = np.random.randn(m, n)
82 # Sample a random orthogonal matrix
83 # by computing eigh of a sampled symmetric matrix.
84 X = np.random.randn(n, n)
85 w, V = eigh(X.T + X)
86 assert_allclose(inv(V), V.T)
87 # Compute a matrix with a known orthogonal transformation that gives B.
88 A = np.dot(B, V.T)
89 # Check that an orthogonal transformation from A to B can be recovered.
90 R, s = orthogonal_procrustes(A, B)
91 assert_allclose(inv(R), R.T)
92 assert_allclose(A.dot(R), B)
93 # Create a perturbed input matrix.
94 A_perturbed = A + 1e-2 * np.random.randn(m, n)
95 # Check that the orthogonal procrustes function can find an orthogonal
96 # transformation that is better than the orthogonal transformation
97 # computed from the original input matrix.
98 R_prime, s = orthogonal_procrustes(A_perturbed, B)
99 assert_allclose(inv(R_prime), R_prime.T)
100 # Compute the naive and optimal transformations of the perturbed input.
101 naive_approx = A_perturbed.dot(R)
102 optim_approx = A_perturbed.dot(R_prime)
103 # Compute the Frobenius norm errors of the matrix approximations.
104 naive_approx_error = norm(naive_approx - B, ord='fro')
105 optim_approx_error = norm(optim_approx - B, ord='fro')
106 # Check that the orthogonal Procrustes approximation is better.
107 assert_array_less(optim_approx_error, naive_approx_error)
