807def _get_optimize(im, info):
808 """
809 Palette optimization is a potentially expensive operation.
810
811 This function determines if the palette should be optimized using
812 some heuristics, then returns the list of palette entries in use.
813
814 :param im: Image object
815 :param info: encoderinfo
816 :returns: list of indexes of palette entries in use, or None
817 """
818 if im.mode in ("P", "L") and info and info.get("optimize", 0):
819 # Potentially expensive operation.
820
821 # The palette saves 3 bytes per color not used, but palette
822 # lengths are restricted to 3*(2**N) bytes. Max saving would
823 # be 768 -> 6 bytes if we went all the way down to 2 colors.
824 # * If we're over 128 colors, we can't save any space.
825 # * If there aren't any holes, it's not worth collapsing.
826 # * If we have a 'large' image, the palette is in the noise.
827
828 # create the new palette if not every color is used
829 optimise = _FORCE_OPTIMIZE or im.mode == "L"
830 if optimise or im.width * im.height < 512 * 512:
831 # check which colors are used
832 used_palette_colors = []
833 for i, count in enumerate(im.histogram()):
834 if count:
835 used_palette_colors.append(i)
836
837 if optimise or max(used_palette_colors) >= len(used_palette_colors):
838 return used_palette_colors
839
840 num_palette_colors = len(im.palette.palette) // Image.getmodebands(
841 im.palette.mode
842 )
843 current_palette_size = 1 << (num_palette_colors - 1).bit_length()
844 if (
845 # check that the palette would become smaller when saved
846 len(used_palette_colors) <= current_palette_size // 2
847 # check that the palette is not already the smallest possible size
