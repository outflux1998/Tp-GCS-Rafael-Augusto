311 """Naive implementation of harmonic sum"""

314 @np.vectorize
315 def pzip(k, a, n):
316 """Naive implementation of zipfian pmf"""
317 if k < 1 or k > n:
318 return 0.
319 else:
320 return 1 / k**a / Hns(n, a)
321
322 k = np.arange(n+1)
323 pmf = pzip(k, a, n)
324 cdf = np.cumsum(pmf)
325 mean = np.average(k, weights=pmf)
326 var = np.average((k - mean)**2, weights=pmf)
327 std = var**0.5
328 skew = np.average(((k-mean)/std)**3, weights=pmf)
329 kurtosis = np.average(((k-mean)/std)**4, weights=pmf) - 3
