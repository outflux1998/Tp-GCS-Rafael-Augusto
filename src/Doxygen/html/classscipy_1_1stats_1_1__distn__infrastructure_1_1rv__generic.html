879 # Subclasses should not have to override this method.
880 # The rule is that if `size` is not None, then `size` gives the
881 # shape of the result (integer values of `size` are treated as
882 # tuples with length 1; i.e. `size=3` is the same as `size=(3,)`.)
883 #
884 # `args` is expected to contain the shape parameters (if any), the
885 # location and the scale in a flat tuple (e.g. if there are two
886 # shape parameters `a` and `b`, `args` will be `(a, b, loc, scale)`).
887 # The only keyword argument expected is 'size'.
888 size = kwargs.get('size', None)
889 all_bcast = np.broadcast_arrays(*args)

891 def squeeze_left(a):
892 while a.ndim > 0 and a.shape[0] == 1:
893 a = a[0]
894 return a

896 # Eliminate trivial leading dimensions. In the convention
897 # used by numpy's random variate generators, trivial leading
898 # dimensions are effectively ignored. In other words, when `size`
899 # is given, trivial leading dimensions of the broadcast parameters
900 # in excess of the number of dimensions in size are ignored, e.g.
901 # >>> np.random.normal([[1, 3, 5]], [[[[0.01]]]], size=3)
902 # array([ 1.00104267, 3.00422496, 4.99799278])
903 # If `size` is not given, the exact broadcast shape is preserved:
904 # >>> np.random.normal([[1, 3, 5]], [[[[0.01]]]])
905 # array([[[[ 1.00862899, 3.00061431, 4.99867122]]]])
906 #
907 all_bcast = [squeeze_left(a) for a in all_bcast]
908 bcast_shape = all_bcast[0].shape

916 # Check compatibility of size_ with the broadcast shape of all
917 # the parameters. This check is intended to be consistent with
918 # how the numpy random variate generators (e.g. np.random.normal,
919 # np.random.beta) handle their arguments. The rule is that, if size
920 # is given, it determines the shape of the output. Broadcasting
921 # can't change the output size.
922
923 # This is the standard broadcasting convention of extending the
924 # shape with fewer dimensions with enough dimensions of length 1
925 # so that the two shapes have the same number of dimensions.

933 # two shapes are compatible if for each dimension, the lengths are the
934 # same or one of the lengths is 1. Here, the length of a dimension in
935 # size_ must not be less than the corresponding length in bcast_shape.
936 ok = all([bcdim == 1 or bcdim == szdim
937 for (bcdim, szdim) in zip(bcast_shape, size_)])
938 if not ok:

940 "the parameters. %s, %s, %s" % (size, size_,
941 bcast_shape))
