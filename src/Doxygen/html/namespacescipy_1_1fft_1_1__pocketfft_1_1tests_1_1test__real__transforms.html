125def naive_dct1(x, norm=None):
126 """Calculate textbook definition version of DCT-I."""
127 x = np.array(x, copy=True)
128 N = len(x)
129 M = N-1
130 y = np.zeros(N)
131 m0, m = 1, 2
132 if norm == 'ortho':
133 m0 = np.sqrt(1.0/M)
134 m = np.sqrt(2.0/M)
135 for k in range(N):
136 for n in range(1, N-1):
137 y[k] += m*x[n]*np.cos(np.pi*n*k/M)
138 y[k] += m0 * x[0]
139 y[k] += m0 * x[N-1] * (1 if k % 2 == 0 else -1)
140 if norm == 'ortho':
141 y[0] *= 1/np.sqrt(2)
142 y[N-1] *= 1/np.sqrt(2)
143 return y
