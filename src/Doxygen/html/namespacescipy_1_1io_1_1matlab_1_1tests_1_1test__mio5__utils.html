33def _make_tag(base_dt, val, mdtype, sde=False):
34 ''' Makes a simple matlab tag, full or sde '''
35 base_dt = np.dtype(base_dt)
36 bo = boc.to_numpy_code(base_dt.byteorder)
37 byte_count = base_dt.itemsize
38 if not sde:
39 udt = bo + 'u4'
40 padding = 8 - (byte_count % 8)
41 all_dt = [('mdtype', udt),
42 ('byte_count', udt),
43 ('val', base_dt)]
44 if padding:
45 all_dt.append(('padding', 'u1', padding))
46 else: # is sde
47 udt = bo + 'u2'
48 padding = 4-byte_count
49 if bo == '<': # little endian
50 all_dt = [('mdtype', udt),
51 ('byte_count', udt),
52 ('val', base_dt)]
53 else: # big endian
54 all_dt = [('byte_count', udt),
55 ('mdtype', udt),
56 ('val', base_dt)]
57 if padding:
58 all_dt.append(('padding', 'u1', padding))
59 tag = np.zeros((1,), dtype=all_dt)
60 tag['mdtype'] = mdtype
61 tag['byte_count'] = byte_count
62 tag['val'] = val
63 return tag
