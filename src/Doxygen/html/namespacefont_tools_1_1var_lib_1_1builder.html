32def VarData_calculateNumShorts(self, optimize=False):
33 count = self.VarRegionCount
34 items = self.Item
35 bit_lengths = [0] * count
36 for item in items:
37 # The "+ (i < -1)" magic is to handle two's-compliment.
38 # That is, we want to get back 7 for -128, whereas
39 # bit_length() returns 8. Similarly for -65536.
40 # The reason "i < -1" is used instead of "i < 0" is that
41 # the latter would make it return 0 for "-1" instead of 1.
42 bl = [(i + (i < -1)).bit_length() for i in item]
43 bit_lengths = [max(*pair) for pair in zip(bl, bit_lengths)]
44 # The addition of 8, instead of seven, is to account for the sign bit.
45 # This "((b + 8) >> 3) if b else 0" when combined with the above
46 # "(i + (i < -1)).bit_length()" is a faster way to compute byte-lengths
47 # conforming to:
48 #
49 # byte_length = (0 if i == 0 else
50 # 1 if -128 <= i < 128 else
51 # 2 if -65536 <= i < 65536 else
52 # ...)
53 byte_lengths = [((b + 8) >> 3) if b else 0 for b in bit_lengths]
54
55 # https://github.com/fonttools/fonttools/issues/2279
56 longWords = any(b > 2 for b in byte_lengths)
57
58 if optimize:
59 # Reorder columns such that wider columns come before narrower columns
60 mapping = []
61 mapping.extend(i for i,b in enumerate(byte_lengths) if b > 2)
62 mapping.extend(i for i,b in enumerate(byte_lengths) if b == 2)
63 mapping.extend(i for i,b in enumerate(byte_lengths) if b == 1)
64
65 byte_lengths = _reorderItem(byte_lengths, mapping)
66 self.VarRegionIndex = _reorderItem(self.VarRegionIndex, mapping)
67 self.VarRegionCount = len(self.VarRegionIndex)
68 for i in range(len(items)):
69 items[i] = _reorderItem(items[i], mapping)
70
71 if longWords:
72 self.NumShorts = max((i for i,b in enumerate(byte_lengths) if b > 2), default=-1) + 1
73 self.NumShorts |= 0x8000
74 else:
75 self.NumShorts = max((i for i,b in enumerate(byte_lengths) if b > 1), default=-1) + 1
76
77 self.VarRegionCount = len(self.VarRegionIndex)
78 return self
