1090 # Ignore ULP differences between various NAN's
1091 # Note that MIPS may reverse quiet and signaling nans
1092 # so we use the builtin version as a base.
1093 offset = np.uint16(0xff)
1094 nan1_i16 = np.array(np.nan, dtype=np.float16).view(np.uint16)
1095 nan2_i16 = nan1_i16 ^ offset # nan payload on MIPS is all ones.

1097 nan2_f16 = nan2_i16.view(np.float16)
1098 assert_array_max_ulp(nan1_f16, nan2_f16, 0)
