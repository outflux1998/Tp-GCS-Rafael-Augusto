65 # Test that sparse the lsqr solver returns the right solution
66 # on various problems with different random seeds.
67 # This is a non-regression test for a potential ZeroDivisionError
68 # raised when computing the `test2` & `test3` convergence conditions.
69 n = 10

71 A_dense = A_sparse.toarray()
72

76 beta = rng.rand(n)
77 beta[beta == 0] = 0.00001 # ensure that all the betas are not null
78 b = A_sparse @ beta[:, np.newaxis]
79 output = lsqr(A_sparse, b, show=show)

82 # solution to Ax = b
