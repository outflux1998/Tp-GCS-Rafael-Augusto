58def test_cluster_size_1():
59 # Assert Silhouette Coefficient == 0 when there is 1 sample in a cluster
60 # (cluster 0). We also test the case where there are identical samples
61 # as the only members of a cluster (cluster 2). To our knowledge, this case
62 # is not discussed in reference material, and we choose for it a sample
63 # score of 1.
64 X = [[0.0], [1.0], [1.0], [2.0], [3.0], [3.0]]
65 labels = np.array([0, 1, 1, 1, 2, 2])
66
67 # Cluster 0: 1 sample -> score of 0 by Rousseeuw's convention
68 # Cluster 1: intra-cluster = [.5, .5, 1]
69 # inter-cluster = [1, 1, 1]
70 # silhouette = [.5, .5, 0]
71 # Cluster 2: intra-cluster = [0, 0]
72 # inter-cluster = [arbitrary, arbitrary]
73 # silhouette = [1., 1.]
74
75 silhouette = silhouette_score(X, labels)
76 assert not np.isnan(silhouette)
77 ss = silhouette_samples(X, labels)
78 assert_array_equal(ss, [0, 0.5, 0.5, 0, 1, 1])
