675 # signed zero can be used, it is checked.

677 tgtdiv = np.where((tgtdiv == 0.0) & ((b < 0) ^ (a < 0)), -0.0, tgtdiv)
678 tgtrem = np.where((tgtrem == 0.0) & (b < 0), -0.0, tgtrem)
