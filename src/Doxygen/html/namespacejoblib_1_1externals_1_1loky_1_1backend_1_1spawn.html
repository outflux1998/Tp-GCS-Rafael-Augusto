221 # Unfortunately, the main ipython launch script historically had no
222 # "if __name__ == '__main__'" guard, so we work around that
223 # by treating it like a __main__.py file
224 # See https://github.com/ipython/ipython/issues/4698
225 main_name = os.path.splitext(os.path.basename(main_path))[0]
226 if main_name == 'ipython':

228
229 # Otherwise, if __file__ already has the setting we expect,
230 # there's nothing more to do

232 return
233

235 # name we assume it is an executable script that may contain
236 # non-main code that needs to be executed
237 old_main_modules.append(current_main)
238 main_module = types.ModuleType("__mp_main__")
239 main_content = runpy.run_path(main_path,
240 run_name="__mp_main__")
