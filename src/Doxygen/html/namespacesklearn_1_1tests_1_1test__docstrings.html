47def filter_errors(errors, method, Klass=None):
48 """
49 Ignore some errors based on the method type.
50
51 These rules are specific for scikit-learn."""
52 for code, message in errors:
53 # We ignore following error code,
54 # - RT02: The first line of the Returns section
55 # should contain only the type, ..
56 # (as we may need refer to the name of the returned
57 # object)
58 # - GL01: Docstring text (summary) should start in the line
59 # immediately after the opening quotes (not in the same line,
60 # or leaving a blank line in between)
61 # - GL02: If there's a blank line, it should be before the
62 # first line of the Returns section, not after (it allows to have
63 # short docstrings for properties).
64
65 if code in ["RT02", "GL01", "GL02"]:
66 continue
67
68 # Ignore PR02: Unknown parameters for properties. We sometimes use
69 # properties for ducktyping, i.e. SGDClassifier.predict_proba
70 # Ignore GL08: Parsing of the method signature failed, possibly because this is
71 # a property. Properties are sometimes used for deprecated attributes and the
72 # attribute is already documented in the class docstring.
73 #
74 # All error codes:
75 # https://numpydoc.readthedocs.io/en/latest/validation.html#built-in-validation-checks
76 if code in ("PR02", "GL08") and Klass is not None and method is not None:
77 method_obj = getattr(Klass, method)
78 if isinstance(method_obj, property):
79 continue
80
81 # Following codes are only taken into account for the
82 # top level class docstrings:
83 # - ES01: No extended summary found
84 # - SA01: See Also section not found
85 # - EX01: No examples section found
86
87 if method is not None and code in ["EX01", "SA01", "ES01"]:
88 continue
89 yield code, message
