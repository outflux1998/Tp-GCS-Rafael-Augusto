160def test_vegalite_to_vega_mimebundle(require_altair_saver, vegalite_spec, vega_spec):
161 # temporay fix for https://github.com/vega/vega-lite/issues/7776
162 def delete_none(axes):
163 for axis in axes:
164 for key, value in list(axis.items()):
165 if value is None:
166 del axis[key]
167 return axes
168
169 bundle = spec_to_mimebundle(
170 spec=vegalite_spec,
171 format="vega",
172 mode="vega-lite",
173 vega_version=alt.VEGA_VERSION,
174 vegalite_version=alt.VEGALITE_VERSION,
175 vegaembed_version=alt.VEGAEMBED_VERSION,
176 )
177
178 bundle["application/vnd.vega.v5+json"]["axes"] = delete_none(
179 bundle["application/vnd.vega.v5+json"]["axes"]
180 )
181 assert bundle == {"application/vnd.vega.v5+json": vega_spec}
