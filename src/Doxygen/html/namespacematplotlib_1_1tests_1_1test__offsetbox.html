196def test_picking(child_type, boxcoords):
197 # These all take up approximately the same area.
198 if child_type == 'draw':
199 picking_child = DrawingArea(5, 5)
200 picking_child.add_artist(mpatches.Rectangle((0, 0), 5, 5, linewidth=0))
201 elif child_type == 'image':
202 im = np.ones((5, 5))
203 im[2, 2] = 0
204 picking_child = OffsetImage(im)
205 elif child_type == 'text':
206 picking_child = TextArea('\N{Black Square}', textprops={'fontsize': 5})
207 else:
208 assert False, f'Unknown picking child type {child_type}'
209
210 fig, ax = plt.subplots()
211 ab = AnnotationBbox(picking_child, (0.5, 0.5), boxcoords=boxcoords)
212 ab.set_picker(True)
213 ax.add_artist(ab)
214
215 calls = []
216 fig.canvas.mpl_connect('pick_event', lambda event: calls.append(event))
217
218 # Annotation should be picked by an event occurring at its center.
219 if boxcoords == 'axes points':
220 x, y = ax.transAxes.transform_point((0, 0))
221 x += 0.5 * fig.dpi / 72
222 y += 0.5 * fig.dpi / 72
223 elif boxcoords == 'axes pixels':
224 x, y = ax.transAxes.transform_point((0, 0))
225 x += 0.5
226 y += 0.5
227 else:
228 x, y = ax.transAxes.transform_point((0.5, 0.5))
229 fig.canvas.draw()
230 calls.clear()
231 MouseEvent(
232 "button_press_event", fig.canvas, x, y, MouseButton.LEFT)._process()
233 assert len(calls) == 1 and calls[0].artist == ab
234
235 # Annotation should *not* be picked by an event at its original center
236 # point when the limits have changed enough to hide the *xy* point.
237 ax.set_xlim(-1, 0)
238 ax.set_ylim(-1, 0)
239 fig.canvas.draw()
240 calls.clear()
241 MouseEvent(
242 "button_press_event", fig.canvas, x, y, MouseButton.LEFT)._process()
243 assert len(calls) == 0
