609 # Test that res.jac.T.dot(res.jac) gives Gauss-Newton approximation
610 # of Hessian. This approximation is computed by doubly differentiating
611 # the cost function and dropping the part containing second derivative
612 # of f. For a scalar function it is computed as
613 # H = (rho' + 2 * rho'' * f**2) * f'**2, if the expression inside the
614 # brackets is less than EPS it is replaced by EPS. Here, we check
615 # against the root of H.
616
617 x = 2.0 # res.x will be this.
618 f = x**2 + 5 # res.fun will be this.
619
620 res = least_squares(fun_trivial, x, jac_trivial, loss='linear',
621 max_nfev=1, method=self.method)
622 assert_equal(res.jac, 2 * x)
623
624 # For `huber` loss the Jacobian correction is identically zero
625 # in outlier region, in such cases it is modified to be equal EPS**0.5.
626 res = least_squares(fun_trivial, x, jac_trivial, loss='huber',
627 max_nfev=1, method=self.method)
628 assert_equal(res.jac, 2 * x * EPS**0.5)
629
630 # Now, let's apply `loss_scale` to turn the residual into an inlier.
631 # The loss function becomes linear.
632 res = least_squares(fun_trivial, x, jac_trivial, loss='huber',
633 f_scale=10, max_nfev=1)
634 assert_equal(res.jac, 2 * x)
635
636 # 'soft_l1' always gives a positive scaling.
637 res = least_squares(fun_trivial, x, jac_trivial, loss='soft_l1',
638 max_nfev=1, method=self.method)
639 assert_allclose(res.jac, 2 * x * (1 + f**2)**-0.75)
640
641 # For 'cauchy' the correction term turns out to be negative, and it
642 # replaced by EPS**0.5.

647 # Now use scaling to turn the residual to inlier.
